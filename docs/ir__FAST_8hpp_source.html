78 void IRsend::sendFAST(uint8_t aCommand, int_fast8_t aNumberOfRepeats) {
79  // Set IR carrier frequency
80  enableIROut(FAST_KHZ); // 38 kHz
81 
82  uint_fast8_t tNumberOfCommands = aNumberOfRepeats + 1;
83  while (tNumberOfCommands > 0) {
84 
85  mark(FAST_HEADER_MARK);
86  space(FAST_HEADER_SPACE);
87 
88  sendPulseDistanceWidthData_P(&FASTProtocolConstants, aCommand | (((uint8_t)(~aCommand)) << 8), FAST_BITS);
89 
90  tNumberOfCommands--;
91  // skip last delay!
92  if (tNumberOfCommands > 0) {
93  // send repeated command in a fixed raster
94  delay(FAST_REPEAT_DISTANCE / MICROS_IN_ONE_MILLI);
95  }
96  }
97 }

99 bool IRrecv::decodeFAST() {
100 
101 // uint_fast8_t tRawlen = decodedIRData.rawlen; // Using a local variable does not improve code size
102 
103  // Check we have the right amount of data (36). The +4 is for initial gap, start bit mark and space + stop bit mark.
104  if (decodedIRData.rawlen != ((2 * FAST_BITS) + 4)) {
105  IR_DEBUG_PRINT(F("FAST: "));
106  IR_DEBUG_PRINT(F("Data length="));
107  IR_DEBUG_PRINT(decodedIRData.rawlen);
108  IR_DEBUG_PRINTLN(F(" is not 36"));
109  return false;
110  }
111 
112  if (!checkHeader_P(&FASTProtocolConstants)) {
113  return false;
114  }
115 
116  if (!decodePulseDistanceWidthData_P(&FASTProtocolConstants, FAST_BITS)) {
117 #if defined(LOCAL_DEBUG)
118  Serial.print(F("FAST: "));
119  Serial.println(F("Decode failed"));
120 #endif
121  return false;
122  }
123 
124  WordUnion tValue;
125  tValue.UWord = decodedIRData.decodedRawData;
126 
127  if (tValue.UByte.LowByte != (uint8_t) ~(tValue.UByte.HighByte)) {
128 #if defined(LOCAL_DEBUG)
129  Serial.print(F("FAST: "));
130  Serial.print(F("8 bit parity is not correct. Expected=0x"));
131  Serial.print((uint8_t)~(tValue.UByte.LowByte), HEX);
132  Serial.print(F(" received=0x"));
133  Serial.print(tValue.UByte.HighByte, HEX);
134  Serial.print(F(" data=0x"));
135  Serial.println(tValue.UWord, HEX);
136 #endif
137  decodedIRData.flags = IRDATA_FLAGS_PARITY_FAILED;
138  }
139 
140  checkForRepeatSpaceTicksAndSetFlag(FAST_MAXIMUM_REPEAT_DISTANCE / MICROS_PER_TICK);
141 
142  // Success
143 // decodedIRData.flags = IRDATA_FLAGS_IS_LSB_FIRST; // Not required, since this is the start value
144  decodedIRData.command = tValue.UByte.LowByte;
145  decodedIRData.address = 0; // No address for this protocol
146  decodedIRData.numberOfBits = FAST_BITS;
147  decodedIRData.protocol = FAST;
148 
149  return true;
150 }
