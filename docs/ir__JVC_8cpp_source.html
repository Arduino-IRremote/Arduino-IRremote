40 //==============================================================================
41 // JJJJJ V V CCCC
42 // J V V C
43 // J V V C
44 // J J V V C
45 // J V CCCC
46 //==============================================================================
47 // https://www.sbprojects.net/knowledge/ir/jvc.php
48 // http://www.hifi-remote.com/johnsfine/DecodeIR.html#JVC
49 // IRP: {38k,525}<1,-1|1,-3>(16,-8,(D:8,F:8,1,-45)+)
50 // LSB first, 1 start bit + 8 bit address + 8 bit command + 1 stop bit.
51 // The JVC protocol repeats by skipping the header mark and space -> this leads to a poor repeat detection for JVC protocol.
52 #define JVC_ADDRESS_BITS 8 // 8 bit address
53 #define JVC_COMMAND_BITS 8 // Command
54 
55 #define JVC_BITS (JVC_ADDRESS_BITS + JVC_COMMAND_BITS) // 16 - The number of bits in the protocol
56 #define JVC_UNIT 526
57 
58 #define JVC_HEADER_MARK (16 * JVC_UNIT) // 8400
59 #define JVC_HEADER_SPACE (8 * JVC_UNIT) // 4200
60 
61 #define JVC_BIT_MARK JVC_UNIT // The length of a Bit:Mark
62 #define JVC_ONE_SPACE (3 * JVC_UNIT) // 1578 - The length of a Bit:Space for 1's
63 #define JVC_ZERO_SPACE JVC_UNIT // The length of a Bit:Space for 0's
64 
65 #define JVC_REPEAT_SPACE (uint16_t)(45 * JVC_UNIT) // 23625 - Commands are repeated with a distance of 23 ms for as long as the key on the remote control is held down.
66 
67 //+=============================================================================
68 // JVC does NOT repeat by sending a separate code (like NEC does).
69 // The JVC protocol repeats by skipping the header.
70 //

72 void IRsend::sendJVC(uint8_t aAddress, uint8_t aCommand, uint_fast8_t aNumberOfRepeats) {
73  // Set IR carrier frequency
74  enableIROut(38);
75 
76  // Header
77  mark(JVC_HEADER_MARK);
78  space(JVC_HEADER_SPACE);
79 
80  uint_fast8_t tNumberOfCommands = aNumberOfRepeats + 1;
81  while (tNumberOfCommands > 0) {
82 
83  // Address + command
84  sendPulseDistanceWidthData(JVC_BIT_MARK, JVC_ONE_SPACE, JVC_BIT_MARK, JVC_ZERO_SPACE,
85  aAddress | (aCommand << JVC_ADDRESS_BITS), JVC_BITS, PROTOCOL_IS_LSB_FIRST, SEND_STOP_BIT);
86 
87  tNumberOfCommands--;
88  // skip last delay!
89  if (tNumberOfCommands > 0) {
90  // send repeated command in a fixed raster
91  delay(JVC_REPEAT_SPACE / 1000);
92  }
93  }
94 }

96 /*
97  * First check for right data length
98  * Next check start bit
99  * Next try the decode
100  */
101 bool IRrecv::decodeJVC() {
102 
103  // Check we have the right amount of data (36 or 34). The +4 is for initial gap, start bit mark and space + stop bit mark. +2 is for repeats
104  if (decodedIRData.rawDataPtr->rawlen != ((2 * JVC_BITS) + 4) && decodedIRData.rawDataPtr->rawlen != ((2 * JVC_BITS) + 2)) {
105  DBG_PRINT(F("JVC: "));
106  DBG_PRINT("Data length=");
107  DBG_PRINT(decodedIRData.rawDataPtr->rawlen);
108  DBG_PRINTLN(" is not 34 or 36");
109  return false;
110  }
111 
112  if (decodedIRData.rawDataPtr->rawlen == ((2 * JVC_BITS) + 2)) {
113  /*
114  * Check for repeat
115  * Check leading space and first and last mark length
116  */
117  if (decodedIRData.rawDataPtr->rawbuf[0] < ((JVC_REPEAT_SPACE + (JVC_REPEAT_SPACE / 2) / MICROS_PER_TICK))
118  && matchMark(decodedIRData.rawDataPtr->rawbuf[1], JVC_BIT_MARK)
119  && matchMark(decodedIRData.rawDataPtr->rawbuf[decodedIRData.rawDataPtr->rawlen - 1], JVC_BIT_MARK)) {
120  /*
121  * We have a repeat here, so do not check for start bit
122  */
123  decodedIRData.flags = IRDATA_FLAGS_IS_REPEAT | IRDATA_FLAGS_IS_LSB_FIRST;
124  decodedIRData.address = lastDecodedAddress;
125  decodedIRData.command = lastDecodedCommand;
126  decodedIRData.protocol = JVC;
127  }
128  } else {
129 
130  // Check header "mark" and "space"
131  if (!matchMark(decodedIRData.rawDataPtr->rawbuf[1], JVC_HEADER_MARK)
132  || !matchSpace(decodedIRData.rawDataPtr->rawbuf[2], JVC_HEADER_SPACE)) {
133  DBG_PRINT("JVC: ");
134  DBG_PRINTLN("Header mark or space length is wrong");
135  return false;
136  }
137 
138  if (!decodePulseDistanceData(JVC_BITS, 3, JVC_BIT_MARK, JVC_ONE_SPACE, JVC_ZERO_SPACE, PROTOCOL_IS_LSB_FIRST)) {
139  DBG_PRINT(F("JVC: "));
140  DBG_PRINTLN(F("Decode failed"));
141  return false;
142  }
143 
144  // Success
145 // decodedIRData.flags = IRDATA_FLAGS_IS_LSB_FIRST; // Not required, since this is the start value
146  uint8_t tCommand = decodedIRData.decodedRawData >> JVC_ADDRESS_BITS; // upper 8 bits of LSB first value
147  uint8_t tAddress = decodedIRData.decodedRawData & 0xFF; // lowest 8 bit of LSB first value
148 
149  decodedIRData.command = tCommand;
150  decodedIRData.address = tAddress;
151  decodedIRData.numberOfBits = JVC_BITS;
152  decodedIRData.protocol = JVC;
153  }
154 
155  return true;
156 }

159 bool IRrecv::decodeJVCMSB(decode_results *aResults) {
160  unsigned int offset = 1; // Skip first space
161 
162  // Check for repeat
163  if ((aResults->rawlen - 1 == 33) && matchMark(aResults->rawbuf[offset], JVC_BIT_MARK)
164  && matchMark(aResults->rawbuf[aResults->rawlen - 1], JVC_BIT_MARK)) {
165  aResults->bits = 0;
166  aResults->value = 0xFFFFFFFF;
167  decodedIRData.flags = IRDATA_FLAGS_IS_REPEAT;
168  decodedIRData.protocol = JVC;
169  return true;
170  }
171 
172  // Initial mark
173  if (!matchMark(aResults->rawbuf[offset], JVC_HEADER_MARK)) {
174  return false;
175  }
176  offset++;
177 
178  // Check we have enough data - +3 for start bit mark and space + stop bit mark
179  if (aResults->rawlen <= (2 * JVC_BITS) + 3) {
180  DBG_PRINT("Data length=");
181  DBG_PRINT(aResults->rawlen);
182  DBG_PRINTLN(" is too small. >= 36 is required.");
183 
184  return false;
185  }
186 
187  // Initial space
188  if (!matchSpace(aResults->rawbuf[offset], JVC_HEADER_SPACE)) {
189  return false;
190  }
191  offset++;
192 
193  if (!decodePulseDistanceData(JVC_BITS, offset, JVC_BIT_MARK, JVC_ONE_SPACE, JVC_ZERO_SPACE, PROTOCOL_IS_MSB_FIRST)) {
194  return false;
195  }
196 
197  // Stop bit
198  if (!matchMark(aResults->rawbuf[offset + (2 * JVC_BITS)], JVC_BIT_MARK)) {
199  DBG_PRINTLN(F("Stop bit mark length is wrong"));
200  return false;
201  }
202 
203  // Success
204  aResults->value = decodedIRData.decodedRawData;
205  aResults->bits = JVC_BITS;
206  aResults->decode_type = JVC;
207  decodedIRData.protocol = JVC;
208 
209  return true;
210 }
