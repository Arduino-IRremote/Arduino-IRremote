167 void printIRResultShort(Print *aSerial, IRData *aIRDataPtr) {
168  if (aIRDataPtr->flags & IRDATA_FLAGS_WAS_OVERFLOW) {
169  aSerial->println(F("Overflow"));
170  return;
171  }
172  aSerial->print(F("Protocol="));
173  aSerial->print(getProtocolString(aIRDataPtr->protocol));
174  if (aIRDataPtr->protocol == UNKNOWN) {
175 #if defined(DECODE_HASH)
176  aSerial->print(F(" Hash=0x"));
177 #if (__INT_WIDTH__ < 32)
178  aSerial->print(aIRDataPtr->decodedRawData, HEX);
179 #else
180  PrintULL::print(aSerial,aIRDataPtr->decodedRawData, HEX);
181 #endif
182 
183 #endif
184 #if !defined(DISABLE_CODE_FOR_RECEIVER)
185  aSerial->print(' ');
186  aSerial->print((aIRDataPtr->rawlen + 1) / 2, DEC);
187  aSerial->println(F(" bits (incl. gap and start) received"));
188 #endif
189  } else {
190 #if defined(DECODE_DISTANCE_WIDTH)
191  if (aIRDataPtr->protocol != PULSE_DISTANCE && aIRDataPtr->protocol != PULSE_WIDTH) {
192 #endif
193  /*
194  * New decoders have address and command
195  */
196  aSerial->print(F(" Address=0x"));
197  aSerial->print(aIRDataPtr->address, HEX);
198 
199  aSerial->print(F(" Command=0x"));
200  aSerial->print(aIRDataPtr->command, HEX);
201 
202  if (aIRDataPtr->flags & IRDATA_FLAGS_EXTRA_INFO) {
203  aSerial->print(F(" Extra=0x"));
204  aSerial->print(aIRDataPtr->extra, HEX);
205  }
206 
207  if (aIRDataPtr->flags & IRDATA_FLAGS_PARITY_FAILED) {
208  aSerial->print(F(" Parity fail"));
209  }
210 
211  if (aIRDataPtr->flags & IRDATA_FLAGS_TOGGLE_BIT) {
212  aSerial->print(F(" Toggle=1"));
213  }
214 #if defined(DECODE_DISTANCE_WIDTH)
215  }
216 #endif
217 
218  /*
219  * Print raw data
220  */
221  if (!(aIRDataPtr->flags & IRDATA_FLAGS_IS_REPEAT) || aIRDataPtr->decodedRawData != 0) {
222  aSerial->print(F(" Raw-Data=0x"));
223 #if (__INT_WIDTH__ < 32)
224  aSerial->print(aIRDataPtr->decodedRawData, HEX);
225 #else
226  PrintULL::print(aSerial, aIRDataPtr->decodedRawData, HEX);
227 #endif
228  /*
229  * Print number of bits processed
230  */
231  aSerial->print(' ');
232  aSerial->print(aIRDataPtr->numberOfBits, DEC);
233  aSerial->print(F(" bits"));
234 
235  if (aIRDataPtr->flags & IRDATA_FLAGS_IS_MSB_FIRST) {
236  aSerial->print(F(" MSB first"));
237  } else {
238  aSerial->print(F(" LSB first"));
239  }
240  }
241 
242  /*
243  * Print gap and duration, in order to be able to compute the repeat period of the protocol by adding the next gap time
244  */
245  if (aIRDataPtr->flags & (IRDATA_FLAGS_IS_AUTO_REPEAT | IRDATA_FLAGS_IS_REPEAT)) {
246  aSerial->print(' ');
247  if (aIRDataPtr->flags & IRDATA_FLAGS_IS_AUTO_REPEAT) {
248  aSerial->print(F("Auto-"));
249  }
250  aSerial->print(F("Repeat"));
251  }
252 #if !defined(DISABLE_CODE_FOR_RECEIVER)
253  aSerial->print(F(" Gap="));
254  aSerial->print((uint32_t) aIRDataPtr->initialGapTicks * MICROS_PER_TICK);
255  aSerial->print(F("us"));
256 
257  uint16_t tSumOfDurationTicks = 0;
258  for (IRRawlenType i = 1; i < aIRDataPtr->rawlen; i++) {
259  tSumOfDurationTicks += aIRDataPtr->rawDataPtr->rawbuf[i];
260  }
261  aSerial->print(F(" Duration="));
262  aSerial->print((uint32_t) tSumOfDurationTicks * MICROS_PER_TICK, DEC);
263  aSerial->println(F("us"));
264 #else
265  aSerial->println();
266 #endif
267  }
268 }

271  * Function to bit reverse OLD MSB values of e.g. NEC.
272  **********************************************************************************************************************/
273 uint8_t bitreverseOneByte(uint8_t aValue) {
274 // uint8_t tReversedValue;
275 // return __builtin_avr_insert_bits(0x01234567, aValue, tReversedValue);
276 // 76543210
277  aValue = (aValue >> 4) | (aValue << 4); // Swap in groups of 4
278 // 32107654
279  aValue = ((aValue & 0xcc) >> 2) | ((aValue & 0x33) << 2); // Swap in groups of 2
280 // 10325476
281  aValue = ((aValue & 0xaa) >> 1) | ((aValue & 0x55) << 1); // Swap bit pairs
282 // 01234567
283  return aValue;
284 }

286 uint32_t bitreverse32Bit(uint32_t aInput) {
287 // __builtin_avr_insert_bits();
288  LongUnion tValue;
289  tValue.UByte.HighByte = bitreverseOneByte(aInput);
290  tValue.UByte.MidHighByte = bitreverseOneByte(aInput >> 8);
291  tValue.UByte.MidLowByte = bitreverseOneByte(aInput >> 16);
292  tValue.UByte.LowByte = bitreverseOneByte(aInput >> 24);
293  return tValue.ULong;
294 }
