414 bool IRrecv::decodePulseWidthData(uint_fast8_t aNumberOfBits, uint_fast8_t aStartOffset, unsigned int aOneMarkMicros,
415  unsigned int aZeroMarkMicros, unsigned int aBitSpaceMicros, bool aMSBfirst) {
416 
417  unsigned int *tRawBufPointer = &decodedIRData.rawDataPtr->rawbuf[aStartOffset];
418  uint32_t tDecodedData = 0;
419 
420  if (aMSBfirst) {
421  /*
422  * MSB first is currently optimized out by the compiler, since it is never used.
423  */
424  for (uint_fast8_t i = 0; i < aNumberOfBits; i++) {
425  // Check for variable length mark indicating a 0 or 1
426  if (matchMark(*tRawBufPointer, aOneMarkMicros)) {
427  tDecodedData = (tDecodedData << 1) | 1;
428  IR_TRACE_PRINT('1');
429  } else if (matchMark(*tRawBufPointer, aZeroMarkMicros)) {
430  tDecodedData = (tDecodedData << 1) | 0;
431  IR_TRACE_PRINT('0');
432  } else {
433  IR_DEBUG_PRINT(F("Mark="));
434  IR_DEBUG_PRINT(*tRawBufPointer * MICROS_PER_TICK);
435  IR_DEBUG_PRINT(F(" is not "));
436  IR_DEBUG_PRINT(aOneMarkMicros);
437  IR_DEBUG_PRINT(F(" or "));
438  IR_DEBUG_PRINT(aZeroMarkMicros);
439  IR_DEBUG_PRINT(' ');
440  return false;
441  }
442  tRawBufPointer++;
443 
444  if (tRawBufPointer < &decodedIRData.rawDataPtr->rawbuf[decodedIRData.rawDataPtr->rawlen]) {
445  // Assume that last space, which is not recorded, is correct, since we can not check it
446  // Check for constant length space
447  if (!matchSpace(*tRawBufPointer, aBitSpaceMicros)) {
448  IR_DEBUG_PRINT(F("Space="));
449  IR_DEBUG_PRINT(*tRawBufPointer * MICROS_PER_TICK);
450  IR_DEBUG_PRINT(F(" is not "));
451  IR_DEBUG_PRINT(aBitSpaceMicros);
452  IR_DEBUG_PRINT(' ');
453  return false;
454  }
455  tRawBufPointer++;
456  }
457  }
458  IR_TRACE_PRINTLN(F(""));
459  } else {
460  // LSB first
461  for (uint32_t tMask = 1UL; aNumberOfBits > 0; tMask <<= 1, aNumberOfBits--) {
462 
463  // Check for variable length mark indicating a 0 or 1
464  if (matchMark(*tRawBufPointer, aOneMarkMicros)) {
465  tDecodedData |= tMask; // set the bit
466  IR_TRACE_PRINT('1');
467  } else if (matchMark(*tRawBufPointer, aZeroMarkMicros)) {
468  // do not set the bit
469  IR_TRACE_PRINT('0');
470  } else {
471  IR_DEBUG_PRINT(F("Mark="));
472  IR_DEBUG_PRINT(*tRawBufPointer * MICROS_PER_TICK);
473  IR_DEBUG_PRINT(F(" is not "));
474  IR_DEBUG_PRINT(aOneMarkMicros);
475  IR_DEBUG_PRINT(F(" or "));
476  IR_DEBUG_PRINT(aZeroMarkMicros);
477  IR_DEBUG_PRINT(' ');
478  return false;
479  }
480  tRawBufPointer++;
481 
482  if (tRawBufPointer < &decodedIRData.rawDataPtr->rawbuf[decodedIRData.rawDataPtr->rawlen]) {
483  // Assume that last space, which is not recorded, is correct, since we can not check it
484  // Check for constant length space
485  if (!matchSpace(*tRawBufPointer, aBitSpaceMicros)) {
486  IR_DEBUG_PRINT(F("Space="));
487  IR_DEBUG_PRINT(*tRawBufPointer * MICROS_PER_TICK);
488  IR_DEBUG_PRINT(F(" is not "));
489  IR_DEBUG_PRINT(aBitSpaceMicros);
490  IR_DEBUG_PRINT(' ');
491  return false;
492  }
493  tRawBufPointer++;
494  }
495  }
496  IR_TRACE_PRINTLN(F(""));
497  }
498  decodedIRData.decodedRawData = tDecodedData;
499  return true;
500 }

622 uint_fast8_t IRrecv::getBiphaselevel() {
623  uint_fast8_t tLevelOfCurrentInterval; // 0 (SPACE) or 1 (MARK)
624 
625  if (sBiphaseDecodeRawbuffOffset >= decodedIRData.rawDataPtr->rawlen) {
626  return SPACE; // After end of recorded buffer, assume space.
627  }
628 
629  tLevelOfCurrentInterval = (sBiphaseDecodeRawbuffOffset) & 1; // on odd rawbuf offsets we have mark timings
630 
631  /*
632  * Setup data if sUsedTimingIntervals is 0
633  */
634  if (sUsedTimingIntervals == 0) {
635  unsigned int tCurrentTimingWith = decodedIRData.rawDataPtr->rawbuf[sBiphaseDecodeRawbuffOffset];
636  unsigned int tMarkExcessCorrection = (tLevelOfCurrentInterval == MARK) ? MARK_EXCESS_MICROS : -MARK_EXCESS_MICROS;
637 
638  if (matchTicks(tCurrentTimingWith, (sBiphaseTimeUnit) + tMarkExcessCorrection)) {
639  sCurrentTimingIntervals = 1;
640  } else if (matchTicks(tCurrentTimingWith, (2 * sBiphaseTimeUnit) + tMarkExcessCorrection)) {
641  sCurrentTimingIntervals = 2;
642  } else if (matchTicks(tCurrentTimingWith, (3 * sBiphaseTimeUnit) + tMarkExcessCorrection)) {
643  sCurrentTimingIntervals = 3;
644  } else {
645  return -1;
646  }
647  }
648 
649  // We use another interval from tCurrentTimingIntervals
650  sUsedTimingIntervals++;
651 
652  // keep track of current timing offset
653  if (sUsedTimingIntervals >= sCurrentTimingIntervals) {
654  // we have used all intervals of current timing, switch to next timing value
655  sUsedTimingIntervals = 0;
656  sBiphaseDecodeRawbuffOffset++;
657  }
658 
659  IR_TRACE_PRINTLN(tLevelOfCurrentInterval);
660 
661  return tLevelOfCurrentInterval;
662 }

704 bool IRrecv::decodeHash() {
705  unsigned long hash = FNV_BASIS_32; // the result is the same no matter if we use a long or unsigned long variable
706 
707 // Require at least 6 samples to prevent triggering on noise
708  if (decodedIRData.rawDataPtr->rawlen < 6) {
709  return false;
710  }
711 #if RAW_BUFFER_LENGTH <= 254 // saves around 75 bytes program memory and speeds up ISR
712  uint_fast8_t i;
713 #else
714  unsigned int i;
715 #endif
716  for (i = 1; (i + 2) < decodedIRData.rawDataPtr->rawlen; i++) {
717  uint_fast8_t value = compare(decodedIRData.rawDataPtr->rawbuf[i], decodedIRData.rawDataPtr->rawbuf[i + 2]);
718  // Add value into the hash
719  hash = (hash * FNV_PRIME_32) ^ value;
720  }
721 
722  decodedIRData.decodedRawData = hash;
723  decodedIRData.numberOfBits = 32;
724  decodedIRData.protocol = UNKNOWN;
725 
726  return true;
727 }

1462  * Duration in ticks of 50 us of alternating SPACE, MARK are recorded in irparams.rawbuf array.
1463  * 'rawlen' counts the number of entries recorded so far.
1464  * First entry is the SPACE between transmissions.
1465  *

1468  * As soon as first MARK arrives in IDLE, gap width is recorded and new logging starts.
