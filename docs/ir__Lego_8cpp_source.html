41 // L EEEEEE EEEE OOOO
42 // L E E O O
43 // L EEEE E EEE O O
44 // L E E E O O
45 // LLLLLL EEEEEE EEEE OOOO
46 //==============================================================================
47 // from LEGO Power Functions RC Manual 26.02.2010 Version 1.20
48 // https://github.com/jurriaan/Arduino-PowerFunctions/raw/master/LEGO_Power_Functions_RC_v120.pdf
49 // https://oberguru.net/elektronik/ir/codes/lego_power_functions_train.lircd.conf
50 //
51 // To ensure correct detection of IR messages six 38 kHz cycles are transmitted as mark.
52 // Low bit consists of 6 cycles of IR and 10 “cycles” of pause,
53 // high bit of 6 cycles IR and 21 “cycles” of pause and start bit of 6 cycles IR and 39 “cycles” of pause.
54 // Low bit range 316 - 526 us
55 // High bit range 526 – 947 us
56 // Start/stop bit range 947 – 1579 us
57 
58 // If tm is the maximum message length (16ms) and Ch is the channel number, then
59 // The delay before transmitting the first message is: (4 – Ch)*tm
60 // The time from start to start for the next 2 messages is: 5*tm
61 // The time from start to start for the following messages is: (6 + 2*Ch)*tm
62 
63 // Supported Devices
64 // LEGO Power Functions IR Receiver 8884
65 
66 // MSB first, 1 start bit + 4 bit channel, 4 bit mode + 4 bit command + 4 bit parity + 1 stop bit.
67 #define LEGO_CHANNEL_BITS 4
68 #define LEGO_MODE_BITS 4
69 #define LEGO_COMMAND_BITS 4
70 #define LEGO_PARITY_BITS 4
71 
72 #define LEGO_BITS (LEGO_CHANNEL_BITS + LEGO_MODE_BITS + LEGO_COMMAND_BITS + LEGO_PARITY_BITS)
73 
74 #define LEGO_HEADER_MARK 158 // 6 cycles
75 #define LEGO_HEADER_SPACE 1026 // 39 cycles
76 
77 #define LEGO_BIT_MARK 158 // 6 cycles
78 #define LEGO_ONE_SPACE 553 // 21 cycles
79 #define LEGO_ZERO_SPACE 263 // 10 cycles
80 
81 #define LEGO_AVERAGE_DURATION 11000 // LEGO_HEADER_MARK + LEGO_HEADER_SPACE + 16 * 600 + 158
82 
83 #define LEGO_AUTO_REPEAT_PERIOD_MIN 110000 // Every frame is auto repeated 5 times.
84 #define LEGO_AUTO_REPEAT_PERIOD_MAX 230000 // space for channel 3
85

93 /*
94  * Here we process the structured data, and call the send raw data function
95  */
96 void IRsend::sendLegoPowerFunctions(uint8_t aChannel, uint8_t aCommand, uint8_t aMode, bool aDoSend5Times) {
97  aChannel &= 0x0F; // allow toggle and escape bits too
98  aCommand &= 0x0F;
99  aMode &= 0x0F;
100  uint8_t tParity = 0xF ^ aChannel ^ aMode ^ aCommand;
101  // send 4 bit channel, 4 bit mode, 4 bit command, 4 bit parity
102  uint16_t tRawData = (((aChannel << LEGO_MODE_BITS) | aMode) << (LEGO_COMMAND_BITS + LEGO_PARITY_BITS))
103  | (aCommand << LEGO_PARITY_BITS) | tParity;
104  sendLegoPowerFunctions(tRawData, aChannel, aDoSend5Times);
105 }

107 void IRsend::sendLegoPowerFunctions(uint16_t aRawData, uint8_t aChannel, bool aDoSend5Times) {
108  enableIROut(38);
109 
110  DBG_PRINT("sendLego aRawData=0x");
111  DBG_PRINTLN(aRawData, HEX);
112 
113  aChannel &= 0x03; // we have 4 channels
114 
115  uint_fast8_t tNumberOfCommands = 1;
116  if (aDoSend5Times) {
117  tNumberOfCommands = 5;
118  }
119 // required for repeat timing, see http://www.hackvandedam.nl/blog/?page_id=559
120  uint8_t tRepeatPeriod = (110 - (LEGO_AVERAGE_DURATION / 1000)) + (aChannel * 40); // from 100 to 220
121 
122  while (tNumberOfCommands > 0) {
123 
124  // Header
125  mark(LEGO_HEADER_MARK);
126  space(LEGO_HEADER_SPACE);
127 
128  sendPulseDistanceWidthData(LEGO_BIT_MARK, LEGO_ONE_SPACE, LEGO_BIT_MARK, LEGO_ZERO_SPACE, aRawData, LEGO_BITS, PROTOCOL_IS_MSB_FIRST,
129  SEND_STOP_BIT);
130 
131  tNumberOfCommands--;
132  // skip last delay!
133  if (tNumberOfCommands > 0) {
134  // send repeated command with a fixed space gap
135  delay(tRepeatPeriod);
136  }
137  }
138 }

140 /*
141  * Mode is stored in the upper nibble of command
142  */
143 bool IRrecv::decodeLegoPowerFunctions() {
144 
145  // Check header "mark"
146  if (!matchMark(decodedIRData.rawDataPtr->rawbuf[1], LEGO_HEADER_MARK)) {
147  // no debug output, since this check is mainly to determine the received protocol
148  return false;
149  }
150 
151  // Check we have enough data - +4 for initial gap, start bit mark and space + stop bit mark
152  if (decodedIRData.rawDataPtr->rawlen != (2 * LEGO_BITS) + 4) {
153  DBG_PRINT("LEGO: ");
154  DBG_PRINT("Data length=");
155  DBG_PRINT(decodedIRData.rawDataPtr->rawlen);
156  DBG_PRINTLN(" is not 36");
157  return false;
158  }
159  // Check header "space"
160  if (!matchSpace(decodedIRData.rawDataPtr->rawbuf[2], LEGO_HEADER_SPACE)) {
161  DBG_PRINT("LEGO: ");
162  DBG_PRINTLN("Header space length is wrong");
163  return false;
164  }
165 
166  if (!decodePulseDistanceData(LEGO_BITS, 3, LEGO_BIT_MARK, LEGO_ONE_SPACE, LEGO_ZERO_SPACE, true)) {
167  DBG_PRINT("LEGO: ");
168  DBG_PRINTLN("Decode failed");
169  return false;
170  }
171 
172  // Stop bit
173  if (!matchMark(decodedIRData.rawDataPtr->rawbuf[3 + (2 * LEGO_BITS)], LEGO_BIT_MARK)) {
174  DBG_PRINT("LEGO: ");
175  DBG_PRINTLN(F("Stop bit mark length is wrong"));
176  return false;
177  }
178 
179  // Success
180  decodedIRData.flags = IRDATA_FLAGS_IS_MSB_FIRST;
181  uint16_t tDecodedValue = decodedIRData.decodedRawData;
182  uint8_t tToggleEscapeChannel = tDecodedValue >> (LEGO_MODE_BITS + LEGO_COMMAND_BITS + LEGO_PARITY_BITS);
183  uint8_t tMode = (tDecodedValue >> (LEGO_COMMAND_BITS + LEGO_PARITY_BITS)) & 0xF;
184  uint8_t tData = (tDecodedValue >> LEGO_PARITY_BITS) & 0xF; // lego calls this field "data"
185  uint8_t tParityReceived = tDecodedValue & 0xF;
186 
187  // This is parity as defined in the specifications
188  // But in some scans I saw 0x9 ^ .. as parity formula
189  uint8_t tParityComputed = 0xF ^ tToggleEscapeChannel ^ tMode ^ tData;
190 
191  // parity check
192  if (tParityReceived != tParityComputed) {
193  DBG_PRINT("LEGO: ");
194  DBG_PRINT("Parity is not correct. expected=0x");
195  DBG_PRINT(tParityComputed, HEX);
196  DBG_PRINT(" received=0x");
197  DBG_PRINT(tParityReceived, HEX);
198  DBG_PRINT(", raw=0x");
199  DBG_PRINT(tDecodedValue, HEX);
200  DBG_PRINT(", 3 nibbles are 0x");
201  DBG_PRINT(tToggleEscapeChannel, HEX);
202  DBG_PRINT(", 0x");
203  DBG_PRINT(tMode, HEX);
204  DBG_PRINT(", 0x");
205  DBG_PRINTLN(tData, HEX);
206  // might not be an error, so just continue
207  decodedIRData.flags = IRDATA_FLAGS_PARITY_FAILED | IRDATA_FLAGS_IS_MSB_FIRST;
208  }
209 
210  /*
211  * Check for autorepeat (should happen 4 times for one press)
212  */
213  if (decodedIRData.rawDataPtr->rawbuf[0] < (LEGO_AUTO_REPEAT_PERIOD_MAX / MICROS_PER_TICK)) {
214  decodedIRData.flags |= IRDATA_FLAGS_IS_AUTO_REPEAT;
215  }
216  decodedIRData.address = tToggleEscapeChannel;
217  decodedIRData.command = tData | tMode << LEGO_COMMAND_BITS;
218  decodedIRData.protocol = LEGO_PF;
219  decodedIRData.numberOfBits = LEGO_BITS;
220 
221  return true;
222 }
