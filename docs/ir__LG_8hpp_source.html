65 // LG originally added by Darryl Smith (based on the JVC protocol)
66 // see: https://github.com/Arduino-IRremote/Arduino-IRremote/tree/master/examples/LGAirConditionerSendDemo
67 // see: https://www.mikrocontroller.net/articles/IRMP_-_english#LGAIR
68 // MSB first, 1 start bit + 8 bit address + 16 bit command + 4 bit checksum + 1 stop bit (28 data bits).
69 // Bit and repeat timing is like NEC
70 // LG2 has different header timing and a shorter bit time
71 /*
72  * LG remote IR-LED measurements: Type AKB 73315611 for air conditioner, Ver1.1 from 2011.03.01
73  * Protocol: LG2
74  * Internal crystal: 4 MHz
75  * Header: 8.9 ms mark 4.15 ms space
76  * Data: 500 / 540 and 500 / 1580;
77  * Clock is not synchronized with gate so you have 19 and sometimes 19 and a spike pulses for mark
78  * Duty: 9 us on 17 us off => around 33 % duty
79  * NO REPEAT: If value like temperature has changed during long press, the last value is send at button release.
80  * If you do a double press, the next value can be sent after around 118 ms. Tested with the fan button.
81 
82  * LG remote IR-LED measurements: Type AKB 75095308 for LG TV
83  * Protocol: NEC!!!
84  * Frequency 37.88 kHz
85  * Header: 9.0 ms mark 4.5 ms space
86  * Data: 560 / 560 and 560 / 1680;
87  * Clock is synchronized with gate, mark always starts with a full period
88  * Duty: 13 us on 13 us off => 50 % duty
89  * Repeat: 110 ms 9.0 ms mark, 2250 us space, 560 stop
90  * LSB first!
91  *
92  * The codes of the LG air conditioner are documented in https://github.com/Arduino-IRremote/Arduino-IRremote/blob/master/ac_LG.cpp
93  */

147 uint32_t IRsend::computeLGRawDataAndChecksum(uint8_t aAddress, uint16_t aCommand) {
148  uint32_t tRawData = ((uint32_t) aAddress << (LG_COMMAND_BITS + LG_CHECKSUM_BITS)) | ((uint32_t) aCommand << LG_CHECKSUM_BITS);
149  /*
150  * My guess of the 4 bit checksum
151  * Addition of all 4 nibbles of the 16 bit command
152  */
153  uint8_t tChecksum = 0;
154  uint16_t tTempForChecksum = aCommand;
155  for (int i = 0; i < 4; ++i) {
156  tChecksum += tTempForChecksum & 0xF; // add low nibble
157  tTempForChecksum >>= 4; // shift by a nibble
158  }
159  return (tRawData | (tChecksum & 0xF));
160 }

176 bool IRrecv::decodeLG() {
177  decode_type_t tProtocol = LG;
178  uint16_t tHeaderSpace = LG_HEADER_SPACE;
179 
180  /*
181  * First check for right data length
182  * Next check start bit
183  * Next try the decode
184  */
185 
186 // Check we have the right amount of data (60). The +4 is for initial gap, start bit mark and space + stop bit mark.
187  if (decodedIRData.rawlen != ((2 * LG_BITS) + 4) && (decodedIRData.rawlen != 4)) {
188  // Is only printed, if LOCAL_DEBUG is defined globally
189  IR_DEBUG_PRINT(F("LG: "));
190  IR_DEBUG_PRINT(F("Data length="));
191  IR_DEBUG_PRINT(decodedIRData.rawlen);
192  IR_DEBUG_PRINTLN(F(" is not 60 or 4"));
193  return false;
194  }
195 
196 // Check header "mark" this must be done for repeat and data
197  if (!matchMark(irparams.rawbuf[1], LG_HEADER_MARK)) {
198  if (matchMark(irparams.rawbuf[1], LG2_HEADER_MARK)) {
199  tProtocol = LG2;
200  tHeaderSpace = LG2_HEADER_SPACE;
201  } else {
202 #if defined(LOCAL_DEBUG)
203  Serial.print(F("LG: "));
204  Serial.println(F("Header mark is wrong"));
205 #endif
206  return false; // neither LG nor LG2 header
207  }
208  }
209 
210 // Check for repeat - here we have another header space length
211  if (decodedIRData.rawlen == 4) {
212  if (matchSpace(irparams.rawbuf[2], LG_REPEAT_HEADER_SPACE) && matchMark(irparams.rawbuf[3], LG_BIT_MARK)) {
213  decodedIRData.flags = IRDATA_FLAGS_IS_REPEAT | IRDATA_FLAGS_IS_MSB_FIRST;
214  decodedIRData.address = lastDecodedAddress;
215  decodedIRData.command = lastDecodedCommand;
216  decodedIRData.protocol = lastDecodedProtocol;
217  return true;
218  }
219 #if defined(LOCAL_DEBUG)
220  Serial.print(F("LG: "));
221  Serial.print(F("Repeat header space is wrong"));
222 #endif
223  return false;
224  }
225 
226 // Check command header space
227  if (!matchSpace(irparams.rawbuf[2], tHeaderSpace)) {
228 #if defined(LOCAL_DEBUG)
229  Serial.print(F("LG: "));
230  Serial.println(F("Header space length is wrong"));
231 #endif
232  return false;
233  }
234 
235  decodePulseDistanceWidthData_P(&LGProtocolConstants, LG_BITS);
236 
237 // Success
238  decodedIRData.flags = IRDATA_FLAGS_IS_MSB_FIRST;
239  decodedIRData.command = (decodedIRData.decodedRawData >> LG_CHECKSUM_BITS) & 0xFFFF;
240  decodedIRData.address = decodedIRData.decodedRawData >> (LG_COMMAND_BITS + LG_CHECKSUM_BITS); // first 8 bit
241 
242  /*
243  * My guess of the checksum
244  */
245  uint8_t tChecksum = 0;
246  uint16_t tTempForChecksum = decodedIRData.command;
247  for (int i = 0; i < 4; ++i) {
248  tChecksum += tTempForChecksum & 0xF; // add low nibble
249  tTempForChecksum >>= 4; // shift by a nibble
250  }
251 // Checksum check
252  if ((tChecksum & 0xF) != (decodedIRData.decodedRawData & 0xF)) {
253 #if defined(LOCAL_DEBUG)
254  Serial.print(F("LG: "));
255  Serial.print(F("4 bit checksum is not correct. expected=0x"));
256  Serial.print(tChecksum, HEX);
257  Serial.print(F(" received=0x"));
258  Serial.print((decodedIRData.decodedRawData & 0xF), HEX);
259  Serial.print(F(" data=0x"));
260  Serial.println(decodedIRData.command, HEX);
261 #endif
262  decodedIRData.flags |= IRDATA_FLAGS_PARITY_FAILED;
263  }
264 
265  decodedIRData.protocol = tProtocol; // LG or LG2
266  decodedIRData.numberOfBits = LG_BITS;
267 
268  return true;
269 }

284 bool IRrecv::decodeLGMSB(decode_results *aResults) {
285  unsigned int offset = 1; // Skip first space
286 
287 // Check we have enough data (60) - +4 for initial gap, start bit mark and space + stop bit mark
288  if (aResults->rawlen != (2 * LG_BITS) + 4) {
289  return false;
290  }
291 
292 // Initial mark/space
293  if (!matchMark(aResults->rawbuf[offset], LG_HEADER_MARK)) {
294  return false;
295  }
296  offset++;
297 
298  if (!matchSpace(aResults->rawbuf[offset], LG_HEADER_SPACE)) {
299  return false;
300  }
301  offset++;
302 
303  decodePulseDistanceWidthData(LG_BITS, offset, LG_BIT_MARK, LG_ONE_SPACE, 0, PROTOCOL_IS_MSB_FIRST);
304 
305 // Stop bit
306  if (!matchMark(aResults->rawbuf[offset + (2 * LG_BITS)], LG_BIT_MARK)) {
307 #if defined(LOCAL_DEBUG)
308  Serial.println(F("Stop bit mark length is wrong"));
309 #endif
310  return false;
311  }
312 
313 // Success
314  aResults->value = decodedIRData.decodedRawData;
315  aResults->bits = LG_BITS;
316  aResults->decode_type = LG;
317  decodedIRData.protocol = LG;
318  return true;
319 }
