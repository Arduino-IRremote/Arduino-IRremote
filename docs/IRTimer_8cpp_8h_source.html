902  TC->CTRLA.reg &= ~TC_CTRLA_ENABLE;
903  // When write-synchronization is ongoing for a register, any subsequent write attempts to this register will be discarded, and an error will be reported.
904  while (TC->STATUS.bit.SYNCBUSY == 1) {
905  } // wait for sync

908  // When writing a 1 to the CTRLA.SWRST bit it will immediately read as 1.

910  while (TC->CTRLA.bit.SWRST) {

914  // Use match mode so that the timer counter resets when the count matches the compare register
915  // Set prescaler to 64
916  TC->CTRLA.reg |= TC_CTRLA_MODE_COUNT16 | TC_CTRLA_WAVEGEN_MFRQ | TC_CTRLA_PRESCALER_DIV64 | TC_CTRLA_ENABLE;
