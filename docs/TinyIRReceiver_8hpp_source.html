1 /*
2  * TinyIRReceiver.hpp
3  *
4  * Receives IR data of NEC protocol using pin change interrupts.
5  * NEC is the protocol of most cheap remote controls for Arduino.
6  *
7  * Parity check is done for address and data.
8  * On a completely received IR command, the user function handleReceivedIRData(uint8_t aAddress, uint8_t aCommand, uint8_t aFlags)
9  * is called in interrupt context but with interrupts being enabled to enable use of delay() etc.
10  * !!!!!!!!!!!!!!!!!!!!!!
11  * Functions called in interrupt context should be running as short as possible,
12  * so if you require longer action, save the data (address + command) and handle them in the main loop.
13  * !!!!!!!!!!!!!!!!!!!!!
14  * aFlags can contain one of IRDATA_FLAGS_EMPTY, IRDATA_FLAGS_IS_REPEAT and IRDATA_FLAGS_PARITY_FAILED bits
15  *
16  * The FAST protocol is a proprietary modified JVC protocol without address, with parity and with a shorter header.
17  * FAST Protocol characteristics:
18  * - Bit timing is like NEC or JVC
19  * - The header is shorter, 3156 vs. 12500
20  * - No address and 16 bit data, interpreted as 8 bit command and 8 bit inverted command,
21  * leading to a fixed protocol length of (6 + (16 * 3) + 1) * 526 = 55 * 526 = 28930 microseconds or 29 ms.
22  * - Repeats are sent as complete frames but in a 50 ms period / with a 21 ms distance.
23  *

55  * For more details see: https://github.com/Arduino-IRremote/Arduino-IRremote#compile-options--macros-for-this-library (scroll down)
56  *
57  * - IR_RECEIVE_PIN The pin number for TinyIRReceiver IR input.
58  * - IR_FEEDBACK_LED_PIN The pin number for TinyIRReceiver feedback LED.
59  * - NO_LED_FEEDBACK_CODE Disables the feedback LED function. Saves 14 bytes program memory.
60  * - DISABLE_PARITY_CHECKS Disable parity checks. Saves 48 bytes of program memory.
61  * - USE_EXTENDED_NEC_PROTOCOL Like NEC, but take the 16 bit address as one 16 bit value and not as 8 bit normal and 8 bit inverted value.
62  * - USE_ONKYO_PROTOCOL Like NEC, but take the 16 bit address and command each as one 16 bit value and not as 8 bit normal and 8 bit inverted value.
63  * - USE_FAST_PROTOCOL Use FAST protocol (no address and 16 bit data, interpreted as 8 bit command and 8 bit inverted command) instead of NEC.
64  * - ENABLE_NEC2_REPEATS Instead of sending / receiving the NEC special repeat code, send / receive the original frame for repeat.
65  * - USE_CALLBACK_FOR_TINY_RECEIVER Call the user provided function "void handleReceivedTinyIRData()" each time a frame or repeat is received.
66  */

164 #if defined(ESP8266) || defined(ESP32)
165 IRAM_ATTR
166 #endif
167 void IRPinChangeInterruptHandler(void) {
168 #if defined(_IR_MEASURE_TIMING) && defined(_IR_TIMING_TEST_PIN)
169  digitalWriteFast(_IR_TIMING_TEST_PIN, HIGH); // 2 clock cycles
170 #endif
171  /*
172  * Save IR input level
173  * Negative logic, true / HIGH means inactive / IR space, LOW / false means IR mark.
174  */
175  uint_fast8_t tIRLevel = digitalReadFast(IR_RECEIVE_PIN);
176 
177 #if !defined(NO_LED_FEEDBACK_CODE) && defined(IR_FEEDBACK_LED_PIN)
178  digitalWriteFast(IR_FEEDBACK_LED_PIN, !tIRLevel);
179 #endif
180 
181  /*
182  * 1. compute microseconds after last change
183  */
184  // Repeats can be sent after a pause, which is longer than 64000 microseconds, so we need a 32 bit value for check of repeats
185  uint32_t tCurrentMicros = micros();
186  uint32_t tMicrosOfMarkOrSpace32 = tCurrentMicros - TinyIRReceiverControl.LastChangeMicros;
187  uint16_t tMicrosOfMarkOrSpace = tMicrosOfMarkOrSpace32;
188 
189  TinyIRReceiverControl.LastChangeMicros = tCurrentMicros;
190 
191  uint8_t tState = TinyIRReceiverControl.IRReceiverState;
192 
193 #if defined(LOCAL_TRACE_STATE_MACHINE)
194  Serial.print(tState);
195  Serial.print(F(" D="));
196  Serial.print(tMicrosOfMarkOrSpace);
197 // Serial.print(F(" I="));
198 // Serial.print(tIRLevel);
199  Serial.print('|');
200 #endif
201 
202  if (tIRLevel == LOW) {
203  /*
204  * We have a mark here
205  */
206  if (tMicrosOfMarkOrSpace > 2 * TINY_RECEIVER_HEADER_MARK) {
207  // timeout -> must reset state machine
208  tState = IR_RECEIVER_STATE_WAITING_FOR_START_MARK;
209  }
210  if (tState == IR_RECEIVER_STATE_WAITING_FOR_START_MARK) {
211  // We are at the beginning of the header mark, check timing at the next transition
212  tState = IR_RECEIVER_STATE_WAITING_FOR_START_SPACE;
213  TinyIRReceiverControl.Flags = IRDATA_FLAGS_EMPTY; // If we do it here, it saves 4 bytes
214 #if defined(LOCAL_TRACE)
215  sMicrosOfGap = tMicrosOfMarkOrSpace32;
216 #endif
217 #if defined(ENABLE_NEC2_REPEATS)
218  // Check for repeat, where full frame is sent again after TINY_RECEIVER_REPEAT_PERIOD ms
219  // Not required for NEC, where repeats are detected by a special header space duration
220  // Must use 32 bit arithmetic here!
221  if (tMicrosOfMarkOrSpace32 < TINY_RECEIVER_MAXIMUM_REPEAT_DISTANCE) {
222  TinyIRReceiverControl.Flags = IRDATA_FLAGS_IS_REPEAT;
223  }
224 #endif
225  }
226 
227  else if (tState == IR_RECEIVER_STATE_WAITING_FOR_FIRST_DATA_MARK) {
228  if (tMicrosOfMarkOrSpace >= lowerValue25Percent(TINY_RECEIVER_HEADER_SPACE)
229  && tMicrosOfMarkOrSpace <= upperValue25Percent(TINY_RECEIVER_HEADER_SPACE)) {
230  /*
231  * We had a valid data header space here -> initialize data
232  */
233  TinyIRReceiverControl.IRRawDataBitCounter = 0;
234 #if (TINY_RECEIVER_BITS > 16)
235  TinyIRReceiverControl.IRRawData.ULong = 0;
236 #else
237  TinyIRReceiverControl.IRRawData.UWord = 0;
238 #endif
239  TinyIRReceiverControl.IRRawDataMask = 1;
240  tState = IR_RECEIVER_STATE_WAITING_FOR_DATA_SPACE;
241 #if !defined(ENABLE_NEC2_REPEATS)
242  // Check for NEC repeat header
243  } else if (tMicrosOfMarkOrSpace >= lowerValue25Percent(NEC_REPEAT_HEADER_SPACE)
244  && tMicrosOfMarkOrSpace <= upperValue25Percent(NEC_REPEAT_HEADER_SPACE)
245  && TinyIRReceiverControl.IRRawDataBitCounter >= TINY_RECEIVER_BITS) {
246  /*
247  * We have a repeat header here and no broken receive before -> set repeat flag
248  */
249  TinyIRReceiverControl.Flags = IRDATA_FLAGS_IS_REPEAT;
250  tState = IR_RECEIVER_STATE_WAITING_FOR_DATA_SPACE;
251 #endif
252  } else {
253  // This parts are optimized by the compiler into jumps to one code :-)
254  // Wrong length -> reset state
255  tState = IR_RECEIVER_STATE_WAITING_FOR_START_MARK;
256  }
257  }
258 
259  else if (tState == IR_RECEIVER_STATE_WAITING_FOR_DATA_MARK) {
260  // Check data space length
261  if (tMicrosOfMarkOrSpace >= lowerValue50Percent(TINY_RECEIVER_ZERO_SPACE)
262  && tMicrosOfMarkOrSpace <= upperValue50Percent(TINY_RECEIVER_ONE_SPACE)) {
263  // We have a valid bit here
264  tState = IR_RECEIVER_STATE_WAITING_FOR_DATA_SPACE;
265  if (tMicrosOfMarkOrSpace >= 2 * TINY_RECEIVER_UNIT) {
266  // we received a 1
267 #if (TINY_RECEIVER_BITS > 16)
268  TinyIRReceiverControl.IRRawData.ULong |= TinyIRReceiverControl.IRRawDataMask;
269 #else
270  TinyIRReceiverControl.IRRawData.UWord |= TinyIRReceiverControl.IRRawDataMask;
271 #endif
272  } else {
273  // we received a 0 - empty code for documentation
274  }
275  // prepare for next bit
276  TinyIRReceiverControl.IRRawDataMask = TinyIRReceiverControl.IRRawDataMask << 1;
277  TinyIRReceiverControl.IRRawDataBitCounter++;
278  } else {
279  // Wrong length -> reset state
280  tState = IR_RECEIVER_STATE_WAITING_FOR_START_MARK;
281  }
282  } else {
283  // error wrong state for the received level, e.g. if we missed one change interrupt -> reset state
284  tState = IR_RECEIVER_STATE_WAITING_FOR_START_MARK;
285  }
286  }
287 
288  else {
289  /*
290  * We have a space here
291  */
292  if (tState == IR_RECEIVER_STATE_WAITING_FOR_START_SPACE) {
293  /*
294  * Check length of header mark here
295  */
296  if (tMicrosOfMarkOrSpace >= lowerValue25Percent(TINY_RECEIVER_HEADER_MARK)
297  && tMicrosOfMarkOrSpace <= upperValue25Percent(TINY_RECEIVER_HEADER_MARK)) {
298  tState = IR_RECEIVER_STATE_WAITING_FOR_FIRST_DATA_MARK;
299  } else {
300  // Wrong length of header mark -> reset state
301  tState = IR_RECEIVER_STATE_WAITING_FOR_START_MARK;
302  }
303  }
304 
305  else if (tState == IR_RECEIVER_STATE_WAITING_FOR_DATA_SPACE) {
306  // Check data mark length
307  if (tMicrosOfMarkOrSpace >= lowerValue50Percent(TINY_RECEIVER_BIT_MARK)
308  && tMicrosOfMarkOrSpace <= upperValue50Percent(TINY_RECEIVER_BIT_MARK)) {
309  /*
310  * We have a valid mark here, check for transmission complete, i.e. the mark of the stop bit
311  */
312  if (TinyIRReceiverControl.IRRawDataBitCounter >= TINY_RECEIVER_BITS
313 #if !defined(ENABLE_NEC2_REPEATS)
314  || (TinyIRReceiverControl.Flags & IRDATA_FLAGS_IS_REPEAT) // Do not check for full length received, if we have a short repeat frame
315 #endif
316  ) {
317  /*
318  * Code complete -> optionally check parity
319  */
320  // Reset state for new start
321  tState = IR_RECEIVER_STATE_WAITING_FOR_START_MARK;
322 
323 #if !defined(DISABLE_PARITY_CHECKS) && (TINY_RECEIVER_ADDRESS_BITS == 16) && TINY_RECEIVER_ADDRESS_HAS_8_BIT_PARITY
324  /*
325  * Check address parity
326  * Address is sent first and contained in the lower word
327  */
328  if (TinyIRReceiverControl.IRRawData.UBytes[0] != (uint8_t)(~TinyIRReceiverControl.IRRawData.UBytes[1])) {
329 #if defined(ENABLE_NEC2_REPEATS)
330  TinyIRReceiverControl.Flags |= IRDATA_FLAGS_PARITY_FAILED; // here we can have the repeat flag already set
331 #else
332  TinyIRReceiverControl.Flags = IRDATA_FLAGS_PARITY_FAILED; // here we do not check anything, if we have a repeat
333 #endif
334  }
335 #endif
336 #if !defined(DISABLE_PARITY_CHECKS) && (TINY_RECEIVER_COMMAND_BITS == 16) && TINY_RECEIVER_COMMAND_HAS_8_BIT_PARITY
337  /*
338  * Check command parity
339  */
340 #if (TINY_RECEIVER_ADDRESS_BITS > 0)
341  if (TinyIRReceiverControl.IRRawData.UBytes[2] != (uint8_t)(~TinyIRReceiverControl.IRRawData.UBytes[3])) {
342 #if defined(ENABLE_NEC2_REPEATS)
343  TinyIRReceiverControl.Flags |= IRDATA_FLAGS_PARITY_FAILED;
344 #else
345  TinyIRReceiverControl.Flags = IRDATA_FLAGS_PARITY_FAILED;
346 #endif
347 # if defined(LOCAL_DEBUG)
348  Serial.print(F("Parity check for command failed. Command="));
349  Serial.print(TinyIRReceiverControl.IRRawData.UBytes[2], HEX);
350  Serial.print(F(" parity="));
351  Serial.println(TinyIRReceiverControl.IRRawData.UBytes[3], HEX);
352 # endif
353 #else
354  // No address, so command and parity are in the lowest bytes
355  if (TinyIRReceiverControl.IRRawData.UBytes[0] != (uint8_t) (~TinyIRReceiverControl.IRRawData.UBytes[1])) {
356  TinyIRReceiverControl.Flags |= IRDATA_FLAGS_PARITY_FAILED;
357 # if defined(LOCAL_DEBUG)
358  Serial.print(F("Parity check for command failed. Command="));
359  Serial.print(TinyIRReceiverControl.IRRawData.UBytes[0], HEX);
360  Serial.print(F(" parity="));
361  Serial.println(TinyIRReceiverControl.IRRawData.UBytes[1], HEX);
362 # endif
363 #endif
364  }
365 #endif
366  /*
367  * Call user provided callback here
368  * The parameter size is dependent of the code variant used in order to save program memory.
369  * We have 6 cases: 0, 8 bit or 16 bit address, each with 8 or 16 bit command
370  */
371 #if !defined(ARDUINO_ARCH_MBED) && !defined(ESP32) // no Serial etc. in callback for ESP -> no interrupt required, WDT is running!
372  interrupts(); // enable interrupts, so delay() etc. works in callback
373 #endif
374  TinyIRReceiverData.justWritten = true;
375  TinyIRReceiverData.Flags = TinyIRReceiverControl.Flags;
376 #if (TINY_RECEIVER_ADDRESS_BITS > 0)
377 # if TINY_RECEIVER_ADDRESS_HAS_8_BIT_PARITY
378  // Here we have 8 bit address
379  TinyIRReceiverData.Address = TinyIRReceiverControl.IRRawData.UBytes[0];
380 # else
381  // Here we have 16 bit address
382  TinyIRReceiverData.Address = TinyIRReceiverControl.IRRawData.UWord.LowWord;
383 # endif
384 # if TINY_RECEIVER_COMMAND_HAS_8_BIT_PARITY
385  // Here we have 8 bit command
386  TinyIRReceiverData.Command = TinyIRReceiverControl.IRRawData.UBytes[2];
387 # else
388  // Here we have 16 bit command
389  TinyIRReceiverData.Command = TinyIRReceiverControl.IRRawData.UWord.HighWord;
390 # endif
391 
392 #else
393  // Here we have NO address
394 # if TINY_RECEIVER_COMMAND_HAS_8_BIT_PARITY
395  // Here we have 8 bit command
396  TinyIRReceiverData.Command = TinyIRReceiverControl.IRRawData.UBytes[0];
397 # else
398  // Here we have 16 bit command
399  TinyIRReceiverData.Command = TinyIRReceiverControl.IRRawData.UWord;
400 # endif
401 #endif
402 #if defined(USE_CALLBACK_FOR_TINY_RECEIVER)
403  handleReceivedTinyIRData();
404 #endif
405 
406  } else {
407  // not finished yet
408  tState = IR_RECEIVER_STATE_WAITING_FOR_DATA_MARK;
409  }
410  } else {
411  // Wrong length -> reset state
412  tState = IR_RECEIVER_STATE_WAITING_FOR_START_MARK;
413  }
414  } else {
415  // error wrong state for the received level, e.g. if we missed one change interrupt -> reset state
416  tState = IR_RECEIVER_STATE_WAITING_FOR_START_MARK;
417  }
418  }
419 
420  TinyIRReceiverControl.IRReceiverState = tState;
421 #ifdef _IR_MEASURE_TIMING
422  digitalWriteFast(_IR_TIMING_TEST_PIN, LOW); // 2 clock cycles
423 #endif
424 }

597 # if defined(USE_INT0)
598  // interrupt on any logical change
599  EICRA |= _BV(ISC00);
