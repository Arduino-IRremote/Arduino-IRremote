2  * TinyIRReceiver.hpp
3  *
4  * Receives IR protocol data of NEC protocol using pin change interrupts.
5  * NEC is the protocol of most cheap remote controls for Arduino.
6  *
7  * Parity check is done for address and data.
8  * On a completely received IR command, the user function handleReceivedIRData(uint8_t aAddress, uint8_t aCommand, uint8_t aFlags)
9  * is called in interrupt context but with interrupts being enabled to enable use of delay() etc.
10  * !!!!!!!!!!!!!!!!!!!!!!
11  * Functions called in interrupt context should be running as short as possible,
12  * so if you require longer action, save the data (address + command) and handle them in the main loop.
13  * !!!!!!!!!!!!!!!!!!!!!

139 void IRPinChangeInterruptHandler(void) {
140 #if defined(_IR_MEASURE_TIMING) && defined(_IR_TIMING_TEST_PIN)
141  digitalWriteFast(_IR_TIMING_TEST_PIN, HIGH); // 2 clock cycles
142 #endif
143  /*
144  * Save IR input level
145  * Negative logic, true / HIGH means inactive / IR space, LOW / false means IR mark.
146  */
147  uint_fast8_t tIRLevel = digitalReadFast(IR_INPUT_PIN);
148 
149 #if !defined(NO_LED_FEEDBACK_CODE) && defined(IR_FEEDBACK_LED_PIN)
150  digitalWriteFast(IR_FEEDBACK_LED_PIN, !tIRLevel);
151 #endif
152 
153  /*
154  * 1. compute microseconds after last change
155  */
156  // Repeats can be sent after a pause, which is longer than 64000 microseconds, so we need a 32 bit value for check of repeats
157  uint32_t tCurrentMicros = micros();
158  uint32_t tMicrosOfMarkOrSpace32 = tCurrentMicros - TinyIRReceiverControl.LastChangeMicros;
159  uint16_t tMicrosOfMarkOrSpace = tMicrosOfMarkOrSpace32;
160 
161  TinyIRReceiverControl.LastChangeMicros = tCurrentMicros;
162 
163  uint8_t tState = TinyIRReceiverControl.IRReceiverState;
164 
165 #if defined(LOCAL_TRACE_STATE_MACHINE)
166  Serial.print(tState);
167  Serial.print(F(" D="));
168  Serial.print(tMicrosOfMarkOrSpace);
169 // Serial.print(F(" I="));
170 // Serial.print(tIRLevel);
171  Serial.print('|');
172 #endif
173 
174  if (tIRLevel == LOW) {
175  /*
176  * We have a mark here
177  */
178  if (tMicrosOfMarkOrSpace > 2 * TINY_HEADER_MARK) {
179  // timeout -> must reset state machine
180  tState = IR_RECEIVER_STATE_WAITING_FOR_START_MARK;
181  }
182  if (tState == IR_RECEIVER_STATE_WAITING_FOR_START_MARK) {
183  // We are at the beginning of the header mark, check timing at the next transition
184  tState = IR_RECEIVER_STATE_WAITING_FOR_START_SPACE;
185  TinyIRReceiverControl.Flags = IRDATA_FLAGS_EMPTY; // If we do it here, it saves 4 bytes
186 #if defined(LOCAL_TRACE)
187  sMicrosOfGap = tMicrosOfMarkOrSpace32;
188 #endif
189 #if !defined(ENABLE_NEC_REPEAT_SUPPORT)
190  // Check for repeat, where full frame is sent again after TINY_REPEAT_PERIOD ms
191  // Not required for NEC, where repeats are detected by a special header space duration
192  // Must use 32 bit arithmetic here!
193  if (tMicrosOfMarkOrSpace32 < TINY_MAXIMUM_REPEAT_DISTANCE) {
194  TinyIRReceiverControl.Flags = IRDATA_FLAGS_IS_REPEAT;
195  }
196 #endif
197  }
198 
199  else if (tState == IR_RECEIVER_STATE_WAITING_FOR_FIRST_DATA_MARK) {
200  if (tMicrosOfMarkOrSpace >= lowerValue25Percent(TINY_HEADER_SPACE)
201  && tMicrosOfMarkOrSpace <= upperValue25Percent(TINY_HEADER_SPACE)) {
202  /*
203  * We have a valid data header space here -> initialize data
204  */
205  TinyIRReceiverControl.IRRawDataBitCounter = 0;
206 #if (TINY_BITS > 16)
207  TinyIRReceiverControl.IRRawData.ULong = 0;
208 #else
209  TinyIRReceiverControl.IRRawData.UWord = 0;
210 #endif
211  TinyIRReceiverControl.IRRawDataMask = 1;
212  tState = IR_RECEIVER_STATE_WAITING_FOR_DATA_SPACE;
213 #if defined(ENABLE_NEC_REPEAT_SUPPORT)
214  // Check for NEC repeat header
215  } else if (tMicrosOfMarkOrSpace >= lowerValue25Percent(NEC_REPEAT_HEADER_SPACE)
216  && tMicrosOfMarkOrSpace <= upperValue25Percent(NEC_REPEAT_HEADER_SPACE)
217  && TinyIRReceiverControl.IRRawDataBitCounter >= TINY_BITS) {
218  /*
219  * We have a repeat header here and no broken receive before -> set repeat flag
220  */
221  TinyIRReceiverControl.Flags = IRDATA_FLAGS_IS_REPEAT;
222  tState = IR_RECEIVER_STATE_WAITING_FOR_DATA_SPACE;
223 #endif
224  } else {
225  // This parts are optimized by the compiler into jumps to one code :-)
226  // Wrong length -> reset state
227  tState = IR_RECEIVER_STATE_WAITING_FOR_START_MARK;
228  }
229  }
230 
231  else if (tState == IR_RECEIVER_STATE_WAITING_FOR_DATA_MARK) {
232  // Check data space length
233  if (tMicrosOfMarkOrSpace >= lowerValue50Percent(TINY_ZERO_SPACE)
234  && tMicrosOfMarkOrSpace <= upperValue50Percent(TINY_ONE_SPACE)) {
235  // We have a valid bit here
236  tState = IR_RECEIVER_STATE_WAITING_FOR_DATA_SPACE;
237  if (tMicrosOfMarkOrSpace >= 2 * TINY_UNIT) {
238  // we received a 1
239 #if (TINY_BITS > 16)
240  TinyIRReceiverControl.IRRawData.ULong |= TinyIRReceiverControl.IRRawDataMask;
241 #else
242  TinyIRReceiverControl.IRRawData.UWord |= TinyIRReceiverControl.IRRawDataMask;
243 #endif
244  } else {
245  // we received a 0 - empty code for documentation
246  }
247  // prepare for next bit
248  TinyIRReceiverControl.IRRawDataMask = TinyIRReceiverControl.IRRawDataMask << 1;
249  TinyIRReceiverControl.IRRawDataBitCounter++;
250  } else {
251  // Wrong length -> reset state
252  tState = IR_RECEIVER_STATE_WAITING_FOR_START_MARK;
253  }
254  } else {
255  // error wrong state for the received level, e.g. if we missed one change interrupt -> reset state
256  tState = IR_RECEIVER_STATE_WAITING_FOR_START_MARK;
257  }
258  }
259 
260  else {
261  /*
262  * We have a space here
263  */
264  if (tState == IR_RECEIVER_STATE_WAITING_FOR_START_SPACE) {
265  /*
266  * Check length of header mark here
267  */
268  if (tMicrosOfMarkOrSpace >= lowerValue25Percent(TINY_HEADER_MARK)
269  && tMicrosOfMarkOrSpace <= upperValue25Percent(TINY_HEADER_MARK)) {
270  tState = IR_RECEIVER_STATE_WAITING_FOR_FIRST_DATA_MARK;
271  } else {
272  // Wrong length of header mark -> reset state
273  tState = IR_RECEIVER_STATE_WAITING_FOR_START_MARK;
274  }
275  }
276 
277  else if (tState == IR_RECEIVER_STATE_WAITING_FOR_DATA_SPACE) {
278  // Check data mark length
279  if (tMicrosOfMarkOrSpace >= lowerValue50Percent(TINY_BIT_MARK)
280  && tMicrosOfMarkOrSpace <= upperValue50Percent(TINY_BIT_MARK)) {
281  /*
282  * We have a valid mark here, check for transmission complete, i.e. the mark of the stop bit
283  */
284  if (TinyIRReceiverControl.IRRawDataBitCounter >= TINY_BITS
285 #if defined(ENABLE_NEC_REPEAT_SUPPORT)
286  || (TinyIRReceiverControl.Flags & IRDATA_FLAGS_IS_REPEAT) // Do not check for full length received, if we have a short repeat frame
287 #endif
288  ) {
289  /*
290  * Code complete -> call callback, no parity check!
291  */
292  // Reset state for new start
293  tState = IR_RECEIVER_STATE_WAITING_FOR_START_MARK;
294 #if !defined(ARDUINO_ARCH_MBED) && !defined(ESP32) // no Serial etc. in callback for ESP -> no interrupt required, WDT is running!
295  interrupts(); // enable interrupts, so delay() etc. works in callback
296 #endif
297 #if !defined(DISABLE_PARITY_CHECKS) && (TINY_ADDRESS_BITS == 16) && TINY_ADDRESS_HAS_8_BIT_PARITY
298  /*
299  * Check address parity
300  * Address is sent first and contained in the lower word
301  */
302  if (TinyIRReceiverControl.IRRawData.UBytes[0] != (uint8_t) (~TinyIRReceiverControl.IRRawData.UBytes[1])) {
303  TinyIRReceiverControl.Flags |= IRDATA_FLAGS_PARITY_FAILED;
304  }
305 #endif
306 #if !defined(DISABLE_PARITY_CHECKS) && (TINY_COMMAND_BITS == 16) && TINY_COMMAND_HAS_8_BIT_PARITY
307  /*
308  * Check command parity
309  */
310 #if (TINY_ADDRESS_BITS > 0)
311  if (TinyIRReceiverControl.IRRawData.UBytes[2] != (uint8_t) (~TinyIRReceiverControl.IRRawData.UBytes[3])) {
312  TinyIRReceiverControl.Flags |= IRDATA_FLAGS_PARITY_FAILED;
313 # if defined(LOCAL_DEBUG)
314  Serial.print(F("Parity check for command failed. Command="));
315  Serial.print(TinyIRReceiverControl.IRRawData.UBytes[2], HEX);
316  Serial.print(F(" parity="));
317  Serial.println(TinyIRReceiverControl.IRRawData.UBytes[3], HEX);
318 # endif
319 #else
320  // No address, so command and parity are in the lowest bytes
321  if (TinyIRReceiverControl.IRRawData.UBytes[0] != (uint8_t) (~TinyIRReceiverControl.IRRawData.UBytes[1])) {
322  TinyIRReceiverControl.Flags |= IRDATA_FLAGS_PARITY_FAILED;
323 # if defined(LOCAL_DEBUG)
324  Serial.print(F("Parity check for command failed. Command="));
325  Serial.print(TinyIRReceiverControl.IRRawData.UBytes[0], HEX);
326  Serial.print(F(" parity="));
327  Serial.println(TinyIRReceiverControl.IRRawData.UBytes[1], HEX);
328 # endif
329 #endif
330  }
331 #endif
332  /*
333  * Call user provided callback here
334  * We have 6 cases: 0, 8 bit or 16 bit address, each with 8 or 16 bit command
335  */
336  handleReceivedTinyIRData(
337 #if (TINY_ADDRESS_BITS > 0)
338 # if TINY_ADDRESS_HAS_8_BIT_PARITY
339  // Here we have 8 bit address
340  TinyIRReceiverControl.IRRawData.UBytes[0],
341 # else
342  // Here we have 16 bit address
343  TinyIRReceiverControl.IRRawData.UWord.LowWord,
344 # endif
345 # if TINY_COMMAND_HAS_8_BIT_PARITY
346  // Here we have 8 bit command
347  TinyIRReceiverControl.IRRawData.UBytes[3],
348 # else
349  // Here we have 16 bit command
350  TinyIRReceiverControl.IRRawData.UWord.HighWord,
351 # endif
352 #else
353 
354  // Here we have NO address
355 # if TINY_COMMAND_HAS_8_BIT_PARITY
356  // Here we have 8 bit command
357  TinyIRReceiverControl.IRRawData.UBytes[0],
358 # else
359  // Here we have 16 bit command
360  TinyIRReceiverControl.IRRawData.UWord,
361 # endif
362 #endif
363  TinyIRReceiverControl.Flags);
364 
365  } else {
366  // not finished yet
367  tState = IR_RECEIVER_STATE_WAITING_FOR_DATA_MARK;
368  }
369  } else {
370  // Wrong length -> reset state
371  tState = IR_RECEIVER_STATE_WAITING_FOR_START_MARK;
372  }
373  } else {
374  // error wrong state for the received level, e.g. if we missed one change interrupt -> reset state
375  tState = IR_RECEIVER_STATE_WAITING_FOR_START_MARK;
376  }
377  }
378 
379  TinyIRReceiverControl.IRReceiverState = tState;
380 #ifdef _IR_MEASURE_TIMING
381  digitalWriteFast(_IR_TIMING_TEST_PIN, LOW); // 2 clock cycles
382 #endif
383 }

508 bool enablePCIInterruptForTinyReceiver() {
509 #if defined(_IR_MEASURE_TIMING) && defined(_IR_TIMING_TEST_PIN)
510  pinModeFast(_IR_TIMING_TEST_PIN, OUTPUT);
511 #endif
512 
513 #if defined(USE_ATTACH_INTERRUPT) || defined(USE_ATTACH_INTERRUPT_DIRECT)
514 # if defined(USE_ATTACH_INTERRUPT)
515 #if defined(NOT_AN_INTERRUPT)
516  if(digitalPinToInterrupt(IR_INPUT_PIN) == NOT_AN_INTERRUPT){
517  return false;
518  }
519 #endif
520  // costs 112 bytes program memory + 4 bytes RAM
521  attachInterrupt(digitalPinToInterrupt(IR_INPUT_PIN), IRPinChangeInterruptHandler, CHANGE);
522 # else
523  // 2.2 us more than version configured with macros and not compatible
524  attachInterrupt(IR_INPUT_PIN, IRPinChangeInterruptHandler, CHANGE); // no extra pin mapping here
525 # endif
526 
527 # if defined(LOCAL_DEBUG_ATTACH_INTERRUPT)
528  Serial.println(F("Use attachInterrupt for pin=" STR(IR_INPUT_PIN)));
529 # endif
530 
531 #else
532 # if defined(LOCAL_DEBUG_ATTACH_INTERRUPT)
533  Serial.println(F("Use static interrupt for pin=" STR(IR_INPUT_PIN)));
534 # endif
535 # if defined(USE_INT0)
536  // interrupt on any logical change
537  EICRA |= _BV(ISC00);
538  // clear interrupt bit
539  EIFR |= 1 << INTF0;
540  // enable interrupt on next change
541  EIMSK |= 1 << INT0;
542 
543 # elif defined(USE_INT1)
544  EICRA |= _BV(ISC10);
545 // clear interrupt bit
546  EIFR |= 1 << INTF1;
547 // enable interrupt on next change
