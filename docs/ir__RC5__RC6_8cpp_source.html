43 //==============================================================================
44 // RRRR CCCC 55555
45 // R R C 5
46 // RRRR C 5555
47 // R R C 5
48 // R R CCCC 5555
49 //==============================================================================
50 //
51 // see: https://www.sbprojects.net/knowledge/ir/rc5.php
52 // https://en.wikipedia.org/wiki/Manchester_code
53 // mark->space => 0
54 // space->mark => 1
55 // MSB first 1 start bit, 1 field bit, 1 toggle bit + 5 bit address + 6 bit command (6 bit command plus one field bit for RC5X), no stop bit
56 // duty factor is 25%,
57 //
58 #define RC5_ADDRESS_BITS 5
59 #define RC5_COMMAND_BITS 6
60 #define RC5_COMMAND_FIELD_BIT 1
61 #define RC5_TOGGLE_BIT 1
62 
63 #define RC5_BITS (RC5_COMMAND_FIELD_BIT + RC5_TOGGLE_BIT + RC5_ADDRESS_BITS + RC5_COMMAND_BITS) // 13
64 
65 #define RC5_UNIT 889 // (32 cycles of 36 kHz)
66 
67 #define MIN_RC5_MARKS ((RC5_BITS + 1) / 2) // 7
68 
69 #define RC5_DURATION (15L * RC5_UNIT) // 13335
70 #define RC5_REPEAT_PERIOD (128L * RC5_UNIT) // 113792
71 #define RC5_REPEAT_SPACE (RC5_REPEAT_PERIOD - RC5_DURATION) // 100 ms

76 void IRsend::sendRC5(uint8_t aAddress, uint8_t aCommand, uint_fast8_t aNumberOfRepeats, bool aEnableAutomaticToggle) {
77  // Set IR carrier frequency
78  enableIROut(36);
79 
80  uint16_t tIRData = ((aAddress & 0x1F) << RC5_COMMAND_BITS);
81 
82  if (aCommand < 0x40) {
83  // set field bit to lower field / set inverted upper command bit
84  tIRData |= 1 << (RC5_TOGGLE_BIT + RC5_ADDRESS_BITS + RC5_COMMAND_BITS);
85  } else {
86  // let field bit zero
87  aCommand &= 0x3F;
88  }
89 
90  tIRData |= aCommand;
91 
92  if (aEnableAutomaticToggle) {
93  if (sLastSendToggleValue == 0) {
94  sLastSendToggleValue = 1;
95  // set toggled bit
96  tIRData |= 1 << (RC5_ADDRESS_BITS + RC5_COMMAND_BITS);
97  } else {
98  sLastSendToggleValue = 0;
99  }
100  }
101 
102  uint_fast8_t tNumberOfCommands = aNumberOfRepeats + 1;
103  while (tNumberOfCommands > 0) {
104 
105  // start bit is sent by sendBiphaseData
106  sendBiphaseData(RC5_UNIT, tIRData, RC5_BITS);
107 
108  tNumberOfCommands--;
109  // skip last delay!
110  if (tNumberOfCommands > 0) {
111  // send repeated command in a fixed raster
112  delay(RC5_REPEAT_SPACE / 1000);
113  }
114  }
115 }

129 bool IRrecv::decodeRC5() {
130  uint8_t tBitIndex;
131  uint32_t tDecodedRawData = 0;
132 
133  // Set Biphase decoding start values
134  initBiphaselevel(1, RC5_UNIT); // Skip gap space
135 
136  // Check we have the right amount of data (11 to 26). The +2 is for initial gap and start bit mark.
137  if (decodedIRData.rawDataPtr->rawlen < MIN_RC5_MARKS + 2 && decodedIRData.rawDataPtr->rawlen > ((2 * RC5_BITS) + 2)) {
138  // no debug output, since this check is mainly to determine the received protocol
139  DBG_PRINT(F("RC5: "));
140  DBG_PRINT("Data length=");
141  DBG_PRINT(decodedIRData.rawDataPtr->rawlen);
142  DBG_PRINTLN(" is not between 11 and 26");
143  return false;
144  }
145 
146 // Check start bit, the first space is included in the gap
147  if (getBiphaselevel() != MARK) {
148  DBG_PRINT(F("RC5: "));
149  DBG_PRINTLN("first getBiphaselevel() is not MARK");
150  return false;
151  }
152 
153  /*
154  * Get data bits - MSB first
155  */
156  for (tBitIndex = 0; sBiphaseDecodeRawbuffOffset < decodedIRData.rawDataPtr->rawlen; tBitIndex++) {
157  uint8_t tStartLevel = getBiphaselevel();
158  uint8_t tEndLevel = getBiphaselevel();
159 
160  if ((tStartLevel == SPACE) && (tEndLevel == MARK)) {
161  // we have a space to mark transition here
162  tDecodedRawData = (tDecodedRawData << 1) | 1;
163  } else if ((tStartLevel == MARK) && (tEndLevel == SPACE)) {
164  // we have a mark to space transition here
165  tDecodedRawData = (tDecodedRawData << 1) | 0;
166  } else {
167  // TRACE_PRINT since I saw this too often
168  DBG_PRINT(F("RC5: "));
169  DBG_PRINTLN(F("Decode failed"));
170  return false;
171  }
172  }
173 
174  // Success
175  decodedIRData.numberOfBits = tBitIndex; // must be RC5_BITS
176 
177  LongUnion tValue;
178  tValue.ULong = tDecodedRawData;
179  decodedIRData.decodedRawData = tDecodedRawData;
180  decodedIRData.command = tValue.UByte.LowByte & 0x3F;
181  decodedIRData.address = (tValue.UWord.LowWord >> RC5_COMMAND_BITS) & 0x1F;
182 
183  // Get the inverted 7. command bit for RC5X, the inverted value is always 1 for RC5 and serves as a second start bit.
184  if ((tValue.UWord.LowWord & (1 << (RC5_TOGGLE_BIT + RC5_ADDRESS_BITS + RC5_COMMAND_BITS))) == 0) {
185  decodedIRData.command += 0x40;
186  }
187 
188  decodedIRData.flags = IRDATA_FLAGS_IS_MSB_FIRST;
189  if (tValue.UByte.MidLowByte & 0x8) {
190  decodedIRData.flags = IRDATA_TOGGLE_BIT_MASK | IRDATA_FLAGS_IS_MSB_FIRST;
191  }
192 
193  // check for repeat
194  if (decodedIRData.rawDataPtr->rawbuf[0] < (RC5_REPEAT_PERIOD / MICROS_PER_TICK)) {
195  decodedIRData.flags |= IRDATA_FLAGS_IS_REPEAT;
196  }
197 
198  decodedIRData.protocol = RC5;
199  return true;
200 }

241 void IRsend::sendRC6(uint32_t aRawData, uint8_t aNumberOfBitsToSend) {
242 // Set IR carrier frequency
243  enableIROut(36);
244 
245 // Header
246  mark(RC6_HEADER_MARK);
247  space(RC6_HEADER_SPACE);
248 
249 // Start bit
250  mark(RC6_UNIT);
251  space(RC6_UNIT);
252 
253 // Data MSB first
254  uint32_t mask = 1UL << (aNumberOfBitsToSend - 1);
255  for (uint_fast8_t i = 1; mask; i++, mask >>= 1) {
256  // The fourth bit we send is the "double width toggle bit"
257  unsigned int t = (i == 4) ? (RC6_UNIT * 2) : (RC6_UNIT);
258  if (aRawData & mask) {
259  mark(t);
260  space(t);
261  } else {
262  space(t);
263  mark(t);
264  }
265  }
266 }

346 bool IRrecv::decodeRC6() {
347  uint8_t tBitIndex;
348  uint32_t tDecodedRawData = 0;
349 
350  // Check we have the right amount of data (). The +3 for initial gap, start bit mark and space
351  if (decodedIRData.rawDataPtr->rawlen < MIN_RC6_MARKS + 3 && decodedIRData.rawDataPtr->rawlen > ((2 * RC6_BITS) + 3)) {
352  DBG_PRINT(F("RC6: "));
353  DBG_PRINT("Data length=");
354  DBG_PRINT(decodedIRData.rawDataPtr->rawlen);
355  DBG_PRINTLN(" is not between 15 and 45");
356  return false;
357  }
358 
359  // Check header "mark" and "space", this must be done for repeat and data
360  if (!matchMark(decodedIRData.rawDataPtr->rawbuf[1], RC6_HEADER_MARK)
361  || !matchSpace(decodedIRData.rawDataPtr->rawbuf[2], RC6_HEADER_SPACE)) {
362  // no debug output, since this check is mainly to determine the received protocol
363  DBG_PRINT(F("RC6: "));
364  DBG_PRINTLN("Header mark or space length is wrong");
365  return false;
366  }
367 
368  // Set Biphase decoding start values
369  initBiphaselevel(3, RC6_UNIT); // Skip gap-space and start-bit mark + space
370 
371 // Process first bit, which is known to be a 1 (mark->space)
372  if (getBiphaselevel() != MARK) {
373  DBG_PRINT(F("RC6: "));
374  DBG_PRINTLN("first getBiphaselevel() is not MARK");
375  return false;
376  }
377  if (getBiphaselevel() != SPACE) {
378  DBG_PRINT(F("RC6: "));
379  DBG_PRINTLN("second getBiphaselevel() is not SPACE");
380  return false;
381  }
382 
383  for (tBitIndex = 0; sBiphaseDecodeRawbuffOffset < decodedIRData.rawDataPtr->rawlen; tBitIndex++) {
384  uint8_t tStartLevel; // start level of coded bit
385  uint8_t tEndLevel; // end level of coded bit
386 
387  tStartLevel = getBiphaselevel();
388  if (tBitIndex == 3) {
389  // Toggle bit is double wide; make sure second half is equal first half
390  if (tStartLevel != getBiphaselevel()) {
391  DBG_PRINT(F("RC6: "));
392  DBG_PRINTLN(F("Toggle mark or space length is wrong"));
393  return false;
394  }
395  }
396 
397  tEndLevel = getBiphaselevel();
398  if (tBitIndex == 3) {
399  // Toggle bit is double wide; make sure second half matches
400  if (tEndLevel != getBiphaselevel()) {
401  DBG_PRINT(F("RC6: "));
402  DBG_PRINTLN(F("Toggle mark or space length is wrong"));
403  return false;
404  }
405  }
406 
407  /*
408  * Determine tDecodedRawData bit value by checking the transition type
409  */
410  if ((tStartLevel == MARK) && (tEndLevel == SPACE)) {
411  // we have a mark to space transition here
412  tDecodedRawData = (tDecodedRawData << 1) | 1; // inverted compared to RC5
413  } else if ((tStartLevel == SPACE) && (tEndLevel == MARK)) {
414  // we have a space to mark transition here
415  tDecodedRawData = (tDecodedRawData << 1) | 0;
416  } else {
417  DBG_PRINT(F("RC6: "));
418  DBG_PRINTLN(F("Decode failed"));
419  // we have no transition here or one level is -1 -> error
420  return false; // Error
421  }
422  }
423 
424 // Success
425  decodedIRData.numberOfBits = tBitIndex;
426 
427  LongUnion tValue;
428  tValue.ULong = tDecodedRawData;
429  decodedIRData.decodedRawData = tDecodedRawData;
430 
431  if (tBitIndex < 36) {
432  // RC6
433  decodedIRData.flags = IRDATA_FLAGS_IS_MSB_FIRST;
434  decodedIRData.command = tValue.UByte.LowByte;
435  decodedIRData.address = tValue.UByte.MidLowByte;
436  // Check for toggle flag
437  if ((tValue.UByte.MidHighByte & 1) != 0) {
438  decodedIRData.flags = IRDATA_TOGGLE_BIT_MASK | IRDATA_FLAGS_IS_MSB_FIRST;
439  }
440  } else {
441  // RC6A
442  decodedIRData.flags = IRDATA_FLAGS_IS_MSB_FIRST | IRDATA_FLAGS_EXTRA_INFO;
443  if ((tValue.UByte.MidLowByte & 0x80) != 0) {
444  decodedIRData.flags = IRDATA_TOGGLE_BIT_MASK | IRDATA_FLAGS_IS_MSB_FIRST | IRDATA_FLAGS_EXTRA_INFO;
445  }
446  tValue.UByte.MidLowByte &= 0x87F; // mask toggle bit
447  decodedIRData.command = tValue.UByte.LowByte;
448  decodedIRData.address = tValue.UByte.MidLowByte;
449  // get extra info
450  decodedIRData.extra = tValue.UWord.HighWord;
451  }
452 
453  // check for repeat, do not check toggle bit yet
454  if (decodedIRData.rawDataPtr->rawbuf[0] < ((RC6_REPEAT_SPACE + (RC6_REPEAT_SPACE / 2)) / MICROS_PER_TICK)) {
455  decodedIRData.flags |= IRDATA_FLAGS_IS_REPEAT;
456  }
457 
458  decodedIRData.protocol = RC6;
459  return true;
460 }

473

489 void IRsend::sendRC5ext(uint8_t addr, uint8_t cmd, bool toggle) {
490 // Set IR carrier frequency
491  enableIROut(36);
492 
493  uint8_t addressBits = 5;
494  uint8_t commandBits = 7;
495 // unsigned long nbits = addressBits + commandBits;
496 
497 // Start
498  mark(RC5_UNIT);
499 
500 // Bit #6 of the command part, but inverted!
501  uint8_t cmdBit6 = (1UL << (commandBits - 1)) & cmd;
502  if (cmdBit6) {
503  // Inverted (1 -> 0 = mark-to-space)
504  mark(RC5_UNIT);
505  space(RC5_UNIT);
506  } else {
507  space(RC5_UNIT);
508  mark(RC5_UNIT);
509  }
510  commandBits--;
511 
512 // Toggle bit
513  static int toggleBit = 1;
514  if (toggle) {
515  if (toggleBit == 0) {
516  toggleBit = 1;
517  } else {
518  toggleBit = 0;
519  }
520  }
521  if (toggleBit) {
522  space(RC5_UNIT);
523  mark(RC5_UNIT);
524  } else {
525  mark(RC5_UNIT);
526  space(RC5_UNIT);
527  }
528 
529 // Address
530  for (uint_fast8_t mask = 1UL << (addressBits - 1); mask; mask >>= 1) {
531  if (addr & mask) {
532  space(RC5_UNIT); // 1 is space, then mark
533  mark(RC5_UNIT);
534  } else {
535  mark(RC5_UNIT);
536  space(RC5_UNIT);
537  }
538  }
539 
540 // Command
541  for (uint_fast8_t mask = 1UL << (commandBits - 1); mask; mask >>= 1) {
542  if (cmd & mask) {
543  space(RC5_UNIT); // 1 is space, then mark
544  mark(RC5_UNIT);
545  } else {
546  mark(RC5_UNIT);
547  space(RC5_UNIT);
548  }
549  }
