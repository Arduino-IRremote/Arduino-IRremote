39 //
40 // Bang & Olufsen
41 //
42 //
43 //==============================================================================
44 // https://www.mikrocontroller.net/attachment/33137/datalink.pdf
45 // https://www.mikrocontroller.net/articles/IRMP_-_english#B&O
46 
47 // This protocol is unusual in two ways:
48 
49 // 1. The carrier frequency is 455 kHz
50 
51 // You can build your own receiver as Bang & Olufsen did (check old schematics) or use a TSOP7000
52 // Vishay stopped producing TSOP7000 since 2009 so you will probably only find counterfeits:
53 // https://www.vishay.com/files/whatsnew/doc/ff_FastFacts_CounterfeitTSOP7000_Dec72018.pdf
54 // It is also likely that you will need an oscilloscope to debug a counterfeit TSOP7000
55 // The specimen used to test this code was very noisy and had a very low output current
56 // A somewhat working fix was to put a 4n7 capacitor across the output and ground followed by a pnp emitter follower
57 // Other examples may require a different treatment
58 // This particular receiver also did receive lower frequencies but rather poorly and with a lower delay than usual
59 // If you need to parallel a receiver with another one you may need to delay the signal to get in phase with the other receiver
60 
61 // 2. A stream of messages can be sent back to back with a new message immediately following the previous stop space
62 // It might be that this only happens over IR and not on the datalink protocol
63 // You can choose to support this or not:
64 
65 // Mode 1: Mode with gaps between frames
66 // Do NOT define ENABLE_BEO_WITHOUT_FRAME_GAP and set RECORD_GAP_MICROS to at least 16000 to accept the unusually long 3. start space
67 // Can only receive single messages. Back to back repeats will result in overflow
68 
69 // Mode 2: Break at start mode
70 // Define ENABLE_BEO_WITHOUT_FRAME_GAP and set RECORD_GAP_MICROS to less than 15000
71 
72 // This treats the 3. start space of 15.5 ms as a gap between 2 messages, which makes decoding easier :-).
73 // The receiving of a transmission will then result in a dummy decode of the first 2 start bits with 0 bits data
74 // followed by a 15.5 ms gap and a data frame with one start bit (originally sent as 4. start bit).
75 // If the receiver is not immediately resumed after the 2 start bit frame, partial second frame will be decoded!
76 // Thus debug printing in the wrong place is very likely to break reception!
77 // Make sure to check the number of bits to filter dummy and incomplete messages.
78 
79 // !!! We assume that the real implementations never set the official first header bit to anything other than 0 !!!
80 // !!! We therefore use 4 start bits instead of the specified 3 and in turn ignore the first header bit of the specification !!!
81 
82 // IR messages are 16 bits long. Datalink messages have different lengths.
83 // This implementation supports up to 40 bits total length split into 8 bit data/command and a header/address of variable length
84 // Header data with more than 16 bits is stored in decodedIRData.extra
85 
86 // B&O is a pulse distance protocol, but it has 3 bit values 0, 1 and (equal/repeat) as well as a special start and trailing bit.
87 //
88 // MSB first, 4 start bits + 8 (to 16?) bit address + 8 bit command + 1 special trailing bit + 1 stop bit.
89 // Address can be longer than 8 bit.
90

109 #define BEO_DATA_BITS 8 // Command or character
110 
111 #define BEO_UNIT 3125 // All timings are in microseconds
112 
113 #define BEO_IR_MARK 200 // The length of a mark in the IR protocol
114 
115 // With decode we see length from 200 to 300 and the 300 leads to errors, if we use 200 as mark length for decode.
116 // And the space value is at least 3125, so we can do a reluctant test for the mark anyway.
117 #define BEO_IR_MARK_FOR_DECODE 250
118 #define BEO_DATALINK_MARK (BEO_UNIT / 2) // The length of a mark in the Datalink protocol
119 
120 #define BEO_PULSE_LENGTH_ZERO BEO_UNIT // The length of a one to zero transition
121 #define BEO_PULSE_LENGTH_EQUAL (2 * BEO_UNIT) // 6250 The length of an equal bit
122 #define BEO_PULSE_LENGTH_ONE (3 * BEO_UNIT) // 9375 The length of a zero to one transition
123 #define BEO_PULSE_LENGTH_TRAILING_BIT (4 * BEO_UNIT) // 12500 The length of the stop bit
124 #define BEO_PULSE_LENGTH_START_BIT (5 * BEO_UNIT) // 15625 The length of the start bit
125 // It is not allowed to send two ones or zeros, you must send a one or zero and a equal instead.

168 /*
169  * @param aBackToBack If true send data back to back, which cannot be decoded if ENABLE_BEO_WITHOUT_FRAME_GAP is NOT defined
170  */
171 void IRsend::sendBangOlufsenRaw(uint32_t aRawData, int_fast8_t aBits, bool aBackToBack) {
172 #if defined(USE_NO_SEND_PWM) || defined(SEND_PWM_BY_TIMER) || BEO_KHZ == 38 // BEO_KHZ == 38 is for unit test which runs the B&O protocol with 38 kHz
173 
174  /*
175  * 455 kHz PWM is currently only supported with SEND_PWM_BY_TIMER defined, otherwise maximum is 180 kHz
176  */
177 # if !defined(USE_NO_SEND_PWM)
178 # if defined(SEND_PWM_BY_TIMER)
179  enableHighFrequencyIROut (BEO_KHZ);
180 # elif (BEO_KHZ == 38)
181  enableIROut (BEO_KHZ); // currently only for unit test
182 # endif
183 # endif
184 
185 // AGC / Start - 3 bits + first constant 0 header bit described in the official documentation
186  if (!aBackToBack) {
187  mark(BEO_IR_MARK);
188  }
189  space(BEO_PULSE_LENGTH_ZERO - BEO_IR_MARK);
190  mark(BEO_IR_MARK);
191  space(BEO_PULSE_LENGTH_ZERO - BEO_IR_MARK);
192  mark(BEO_IR_MARK);
193  space(BEO_PULSE_LENGTH_START_BIT - BEO_IR_MARK);
194 
195 // First bit of header is assumed to be a constant 0 to have a fixed state to begin with the equal decisions.
196 // So this first 0 is treated as the last bit of AGC
197  mark(BEO_IR_MARK);
198  space(BEO_PULSE_LENGTH_ZERO - BEO_IR_MARK);
199  bool tLastBitValueWasOne = false;
200 
201 // Header / Data
202  uint32_t mask = 1UL << (aBits - 1);
203  for (; mask; mask >>= 1) {
204  if (tLastBitValueWasOne && !(aRawData & mask)) {
205  mark(BEO_IR_MARK);
206  space(BEO_PULSE_LENGTH_ZERO - BEO_IR_MARK);
207  tLastBitValueWasOne = false;
208  } else if (!tLastBitValueWasOne && (aRawData & mask)) {
209  mark(BEO_IR_MARK);
210  space(BEO_PULSE_LENGTH_ONE - BEO_IR_MARK);
211  tLastBitValueWasOne = true;
212  } else {
213  mark(BEO_IR_MARK);
214  space(BEO_PULSE_LENGTH_EQUAL - BEO_IR_MARK);
215  }
216  }
217 
218 // Stop
219  mark(BEO_IR_MARK);
220  space(BEO_PULSE_LENGTH_TRAILING_BIT - BEO_IR_MARK);
221  mark(BEO_IR_MARK);
222 
223 #else
224  (void) aRawData;
225  (void) aBits;
226  (void) aBackToBack;
227 #endif
228 }

292 #define BEO_MATCH_DELTA (BEO_UNIT / 2 - MICROS_PER_TICK)
293 static bool matchBeoLength(uint16_t aMeasuredTicks, uint16_t aMatchValueMicros) {
294  const uint16_t tMeasuredMicros = aMeasuredTicks * MICROS_PER_TICK;
295  return aMatchValueMicros - BEO_MATCH_DELTA < tMeasuredMicros && tMeasuredMicros < aMatchValueMicros + BEO_MATCH_DELTA;
296 }

298 bool IRrecv::decodeBangOlufsen() {
299 #if defined(ENABLE_BEO_WITHOUT_FRAME_GAP)
300  if (decodedIRData.rawlen != 6 && decodedIRData.rawlen < 36) { // 16 bits minimum
301  BEO_DEBUG_PRINT(F("B&O: "));
302  BEO_DEBUG_PRINT(F("Data length="));
303  BEO_DEBUG_PRINT(decodedIRData.rawlen);
304  BEO_DEBUG_PRINTLN(F(" is not < 36 or 6"));
305 #else
306  if (decodedIRData.rawlen < 44) { // 16 bits minimum
307  BEO_DEBUG_PRINT(F("B&O: "));
308  BEO_DEBUG_PRINT(F("Data length="));
309  BEO_DEBUG_PRINT(decodedIRData.rawlen);
310  BEO_DEBUG_PRINTLN(F(" is not < 44"));
311 #endif
312  return false;
313  }
314 
315 #if defined(SUPPORT_BEO_DATALINK_TIMING_FOR_DECODE)
316  uint16_t protocolMarkLength = 0; // contains BEO_IR_MARK or BEO_DATALINK_MARK depending of 4. mark received
317  uint64_t tDecodedRawData = 0;
318 #else
319  uint32_t tDecodedRawData = 0;
320 #endif
321  uint8_t tLastDecodedBitValue = 0; // the last start bit is assumed to be zero
322  uint8_t tPulseNumber = 0;
323  uint8_t tBitNumber = 0;
324 
325  BEO_TRACE_PRINT(F("Pre gap: "));
326  BEO_TRACE_PRINT((uint32_t)decodedIRData.initialGapTicks * 50);
327  BEO_TRACE_PRINT(F(" raw len: "));
328  BEO_TRACE_PRINTLN(decodedIRData.rawlen);
329 
330 #if defined(ENABLE_BEO_WITHOUT_FRAME_GAP)
331  /*
332  * Check if we have the AGC part of the first frame, i.e. start bit 1 and 2.
333  */
334  if (decodedIRData.rawlen == 6) {
335  if ((matchMark(irparams.rawbuf[3], BEO_IR_MARK_FOR_DECODE)
336  || matchMark(irparams.rawbuf[3], BEO_DATALINK_MARK))
337  && (matchSpace(irparams.rawbuf[4], BEO_PULSE_LENGTH_ZERO - BEO_IR_MARK_FOR_DECODE)
338  || matchSpace(irparams.rawbuf[4], BEO_PULSE_LENGTH_ZERO - BEO_DATALINK_MARK))) {
339  BEO_TRACE_PRINTLN(F("B&O: AGC only part (start bits 1 + 2 of 4) detected"));
340  } else {
341  return false; // no B&O protocol
342  }
343  } else {
344  /*
345  * Check if leading gap is trailing bit of first AGC frame
346  */
347  if (!matchSpace(decodedIRData.initialGapTicks, BEO_PULSE_LENGTH_START_BIT)) {
348  BEO_TRACE_PRINT(F("B&O: Leading gap of ")); // Leading gap is trailing bit of first frame
349  BEO_TRACE_PRINT((uint32_t)decodedIRData.initialGapTicks * 50); // Leading gap is trailing bit of first frame
350  BEO_TRACE_PRINTLN(F(" us is wrong")); // Leading gap is trailing bit of first frame
351  return false; // no B&O protocol
352  }
353 
354  if (matchMark(irparams.rawbuf[1], BEO_IR_MARK_FOR_DECODE)) {
355 # if defined(SUPPORT_BEO_DATALINK_TIMING_FOR_DECODE)
356  protocolMarkLength = BEO_IR_MARK_FOR_DECODE;
357  } else if (matchMark(irparams.rawbuf[1], BEO_DATALINK_MARK)) {
358  protocolMarkLength = BEO_DATALINK_MARK;
359 # endif
360  } else {
361  BEO_TRACE_PRINTLN(F("B&O: mark length is wrong"));
362  return false;
363  }
364 
365  // skip first zero header bit
366  for (uint8_t tRawBufferMarkIndex = 3; tRawBufferMarkIndex < decodedIRData.rawlen; tRawBufferMarkIndex += 2) {
367 #else
368  for (uint8_t tRawBufferMarkIndex = 1; tRawBufferMarkIndex < decodedIRData.rawlen; tRawBufferMarkIndex += 2) {
369 #endif // defined(ENABLE_BEO_WITHOUT_FRAME_GAP)
370 
371  uint16_t markLength = irparams.rawbuf[tRawBufferMarkIndex];
372  uint16_t spaceLength = irparams.rawbuf[tRawBufferMarkIndex + 1];
373 
374  BEO_TRACE_PRINT(tPulseNumber);
375  BEO_TRACE_PRINT(' ');
376  BEO_TRACE_PRINT(markLength * MICROS_PER_TICK);
377  BEO_TRACE_PRINT(' ');
378  BEO_TRACE_PRINT(spaceLength * MICROS_PER_TICK);
379  BEO_TRACE_PRINT(F(" ("));
380  BEO_TRACE_PRINT((markLength + spaceLength) * MICROS_PER_TICK);
381  BEO_TRACE_PRINTLN(F(") "));
382 
383 #if !defined(ENABLE_BEO_WITHOUT_FRAME_GAP)
384  /*
385  * Handle the first 4 start bits
386  * Check if the 3. bit is the long start bit. If we see the long start bit earlier, synchronize bit counter.
387  */
388  if (tPulseNumber < 4) {
389  if (tPulseNumber < 2) {
390  // bit 0 and 1
391  if (matchSpace(spaceLength, BEO_PULSE_LENGTH_START_BIT - BEO_IR_MARK_FOR_DECODE)) {
392  BEO_TRACE_PRINTLN(F(": detected long start bit -> synchronize state now"));
393  tPulseNumber = 2;
394  }
395  } else {
396  if (tPulseNumber == 3) {
397  if (matchMark(markLength, BEO_IR_MARK_FOR_DECODE)) {
398 # if defined(SUPPORT_BEO_DATALINK_TIMING_FOR_DECODE)
399  protocolMarkLength = BEO_IR_MARK_FOR_DECODE;
400  } else if (matchMark(markLength, BEO_DATALINK_MARK)) {
401  protocolMarkLength = BEO_DATALINK_MARK;
402 # endif
403  } else {
404  BEO_DEBUG_PRINTLN(F("B&O: 4. (start) mark length is wrong"));
405  return false;
406  }
407  }
408  // bit 2 and 3
409  if (!matchBeoLength(markLength + spaceLength,
410  (tPulseNumber == 2) ? BEO_PULSE_LENGTH_START_BIT : BEO_PULSE_LENGTH_ZERO)) {
411  BEO_DEBUG_PRINTLN(F("B&O: Start length is wrong"));
412  return false;
413  }
414  }
415  } else {
416 #endif // !defined(ENABLE_BEO_WITHOUT_FRAME_GAP)
417 
418  /*
419  * Decode header / data
420  * First check for length of mark
421  */
422 #if defined(SUPPORT_BEO_DATALINK_TIMING_FOR_DECODE)
423  if (!matchMark(markLength, protocolMarkLength)) {
424 #else
425  if (!matchMark(markLength, BEO_IR_MARK_FOR_DECODE)) {
426 #endif
427  BEO_DEBUG_PRINTLN(F("B&O: Mark length is wrong"));
428  return false;
429  }
430 
431  /*
432  * Check for stop after receiving at least 8 bits for data and 4 bits for header
433  */
434  if (tBitNumber > BEO_DATA_BITS + 4) {
435  if (matchBeoLength(markLength + spaceLength, BEO_PULSE_LENGTH_TRAILING_BIT)) {
436  BEO_DEBUG_PRINTLN(F("B&O: Trailing bit detected"));
437  break;
438  }
439 #if !defined(ENABLE_BEO_WITHOUT_FRAME_GAP)
440  if (tRawBufferMarkIndex >= decodedIRData.rawlen - 3) { // (rawlen - 3) is index of trailing bit mark
441  BEO_DEBUG_PRINTLN(F("B&O: End of buffer, but no trailing bit detected"));
442  return false;
443  }
444 #endif
445  }
446 
447  /*
448  * Decode bit
449  */
450  if (tLastDecodedBitValue == 0 && matchBeoLength(markLength + spaceLength, BEO_PULSE_LENGTH_ONE)) {
451  tLastDecodedBitValue = 1;
452  } else if (tLastDecodedBitValue == 1 && matchBeoLength(markLength + spaceLength, BEO_PULSE_LENGTH_ZERO)) {
453  tLastDecodedBitValue = 0;
454  } else if (!matchBeoLength(markLength + spaceLength, BEO_PULSE_LENGTH_EQUAL)) {
455  BEO_DEBUG_PRINT(F("B&O: Index="));
456  BEO_DEBUG_PRINT(tRawBufferMarkIndex);
457  BEO_DEBUG_PRINT(F(" Length "));
458  BEO_DEBUG_PRINT((markLength + spaceLength) * MICROS_PER_TICK);
459  BEO_DEBUG_PRINTLN(F(" is wrong"));
460  return false;
461  }
462  tDecodedRawData <<= 1;
463  tDecodedRawData |= tLastDecodedBitValue;
464  ++tBitNumber;
465  BEO_TRACE_PRINT(F("Bits "));
466  BEO_TRACE_PRINT(tBitNumber);
467  BEO_TRACE_PRINT(F(" "));
468  BEO_TRACE_PRINT(uint32_t(tDecodedRawData >> BEO_DATA_BITS), HEX);
469  BEO_TRACE_PRINT(F(" "));
470  BEO_TRACE_PRINTLN(uint8_t(tDecodedRawData & ((1 << BEO_DATA_BITS) - 1)), HEX);
471  // End of bit decode
472 #if !defined(ENABLE_BEO_WITHOUT_FRAME_GAP)
473  }
474 
475 #else
476  /*
477  * Check for last bit after decoding it
478  */
479  if (tRawBufferMarkIndex >= decodedIRData.rawlen - 3) { // (rawlen - 3) is index of last bit mark
480  BEO_TRACE_PRINTLN(F("B&O: Last bit reached"));
481  break;
482  }
483 #endif
484 
485  ++tPulseNumber;
486  }
487 #if defined(ENABLE_BEO_WITHOUT_FRAME_GAP)
488  }
489 #endif
490 
491  decodedIRData.protocol = BANG_OLUFSEN;
492  decodedIRData.address = tDecodedRawData >> BEO_DATA_BITS; // lower header tBitNumber
493  decodedIRData.command = tDecodedRawData & ((1 << BEO_DATA_BITS) - 1); // lower 8 tBitNumber
494  decodedIRData.extra = tDecodedRawData >> (BEO_DATA_BITS + 16); // upper header tBitNumber
495  decodedIRData.numberOfBits = tBitNumber;
496  decodedIRData.flags = IRDATA_FLAGS_IS_MSB_FIRST;
497  decodedIRData.decodedRawData = tDecodedRawData;
498 
499  return true;
500 }
